141  int solve(const Operator<D> &A,
142  Vector<D> & x,
143  const Vector<D> & b,
144  const Operator<D> *Mr = nullptr,
145  bool output = false,
146  std::ostream & os = std::cout) const override
147  {
148  Vector<D> resid = b.getZeroClone();
149 
150  A.apply(x, resid);
151  resid.scaleThenAdd(-1, b);
152 
153  Vector<D> initial_guess = x;
154  x.set(0);
155 
156  double r0_norm = b.twoNorm();
157  Vector<D> rhat = resid;
158  Vector<D> p = resid;
159  Vector<D> ap = b.getZeroClone();
160  Vector<D> as = b.getZeroClone();
161 
162  Vector<D> s = x.getZeroClone();
163  double rho = rhat.dot(resid);
164 
165  int num_its = 0;
166  double residual = resid.twoNorm() / r0_norm;
167  if (output) {
168  char buf[100];
169  sprintf(buf, "%5d %16.8e\n", num_its, residual);
170  os << std::string(buf);
171  }
172  if (r0_norm == 0) {
173  return num_its;
174  }
175  while (residual > tolerance && num_its < max_iterations) {
176  if (timer) {
177  timer->start("Iteration");
178  }
179 
180  if (rho == 0) {
181  throw BreakdownError("BiCGStab broke down, rho was 0 on iteration " + std::to_string(num_its));
182  }
183 
184  applyWithPreconditioner(nullptr, A, Mr, p, ap);
185  double alpha = rho / rhat.dot(ap);
186  s.copy(resid);
187  s.addScaled(-alpha, ap);
188  if (s.twoNorm() / r0_norm <= tolerance) {
189  x.addScaled(alpha, p);
190  if (timer) {
191  timer->stop("Iteration");
192  }
193  break;
194  }
195  applyWithPreconditioner(nullptr, A, Mr, s, as);
196  double omega = as.dot(s) / as.dot(as);
197  x.addScaled(alpha, p, omega, s);
198  resid.addScaled(-alpha, ap);
199  resid.addScaled(-omega, as);
200 
201  double rho_new = resid.dot(rhat);
202  double beta = rho_new * alpha / (rho * omega);
203  p.addScaled(-omega, ap);
204  p.scaleThenAdd(beta, resid);
205 
206  num_its++;
207  rho = rho_new;
208  residual = resid.twoNorm() / r0_norm;
209 
210  if (output) {
211  char buf[100];
212  sprintf(buf, "%5d %16.8e\n", num_its, residual);
213  os << std::string(buf);
214  }
215  if (timer) {
216  timer->stop("Iteration");
217  }
218  }
219  if (Mr != nullptr) {
220  Mr->apply(x, resid);
221  x.copy(resid);
222  }
223  x.add(initial_guess);
224  return num_its;
225  }
