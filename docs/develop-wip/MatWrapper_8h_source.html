60  void copyToPetscVec(const Vector<D> &vec, Vec petsc_vec) const
61  {
62  double *petsc_vec_view;
63  size_t curr_index = 0;
64  VecGetArray(petsc_vec, &petsc_vec_view);
65  for (int i = 0; i < vec.getNumLocalPatches(); i++) {
66  for (int c = 0; c < vec.getNumComponents(); c++) {
67  const ComponentView<const double, D> ld = vec.getComponentView(c, i);
68  nested_loop<D>(ld.getStart(), ld.getEnd(), [&](const std::array<int, D> &coord) {
69  petsc_vec_view[curr_index] = ld[coord];
70  curr_index++;
71  });
72  }
73  }
74  VecRestoreArray(petsc_vec, &petsc_vec_view);
75  }
82  void copyToVec(Vec petsc_vec, Vector<D> &vec) const
83  {
84  const double *petsc_vec_view;
85  size_t curr_index = 0;
86  VecGetArrayRead(petsc_vec, &petsc_vec_view);
87  for (int i = 0; i < vec.getNumLocalPatches(); i++) {
88  for (int c = 0; c < vec.getNumComponents(); c++) {
89  ComponentView<double, D> ld = vec.getComponentView(c, i);
90  nested_loop<D>(ld.getStart(), ld.getEnd(), [&](const std::array<int, D> &coord) {
91  ld[coord] = petsc_vec_view[curr_index];
92  curr_index++;
93  });
94  }
95  }
96  VecRestoreArrayRead(petsc_vec, &petsc_vec_view);
97  }
