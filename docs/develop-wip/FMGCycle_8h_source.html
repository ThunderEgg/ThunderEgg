41  void v_visit(const Level<D> &level, const Vector<D> &f, Vector<D> &u) const
42  {
43  if (level.coarsest()) {
44  for (int i = 0; i < num_coarse_sweeps; i++) {
45  level.getSmoother().smooth(f, u);
46  }
47  } else {
48  for (int i = 0; i < num_pre_sweeps; i++) {
49  level.getSmoother().smooth(f, u);
50  }
51 
52  Vector<D> coarser_f = this->restrict(level, f, u);
53 
54  const Level<D> &coarser_level = *level.getCoarser();
55  Vector<D> coarser_u = coarser_f.getZeroClone();
56 
57  this->visit(coarser_level, coarser_f, coarser_u);
58 
59  coarser_level.getInterpolator().interpolate(coarser_u, u);
60 
61  for (int i = 0; i < num_post_sweeps; i++) {
62  level.getSmoother().smooth(f, u);
63  }
64  }
65  }

67  protected:

75  {
76  if (level.coarsest()) {
77  for (int i = 0; i < num_coarse_sweeps; i++) {
78  level.getSmoother().smooth(f, u);

83  }
84 
85  Vector<D> coarser_f = this->restrict(level, f, u);

99
