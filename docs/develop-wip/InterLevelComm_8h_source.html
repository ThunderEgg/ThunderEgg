117  InterLevelComm(const Domain<D> &coarser_domain, const Domain<D> &finer_domain)
118  : coarser_domain(coarser_domain),
119  finer_domain(finer_domain),
120  ns(finer_domain.getNs()),
121  num_ghost_cells(finer_domain.getNumGhostCells())
122  {
123  int my_patch_size = 1;
124  for (size_t axis = 0; axis < D; axis++) {
125  my_patch_size *= ns[axis] + 2 * num_ghost_cells;
126  }
127  patch_size = my_patch_size;
128 
129  // sort into patches with local parents and patches with ghost parents
130  std::deque<std::pair<int, std::reference_wrapper<const PatchInfo<D>>>> local_parents;
131  std::deque<std::reference_wrapper<const PatchInfo<D>>> ghost_parents;
132  std::set<int> ghost_parents_ids;
133 
134  int rank;
135  MPI_Comm_rank(MPI_COMM_WORLD, &rank);
136  // TODO this has to be changed when domain class is updated
137  std::map<int, int> coarser_domain_id_to_local_index_map;
138  for (const PatchInfo<D> &pinfo : this->coarser_domain.getPatchInfoVector()) {
139  coarser_domain_id_to_local_index_map[pinfo.id] = pinfo.local_index;
140  }
141  for (const PatchInfo<D> &patch : this->finer_domain.getPatchInfoVector()) {
142  if (patch.parent_rank == rank) {
143  local_parents.emplace_back(coarser_domain_id_to_local_index_map[patch.parent_id], patch);
144  } else {
145  ghost_parents.push_back(patch);
146  ghost_parents_ids.insert(patch.parent_id);
147  }
148  }
149  num_ghost_patches = ghost_parents_ids.size();
150 
151  // fill in local vector
152  patches_with_local_parent
153  = std::vector<std::pair<int, std::reference_wrapper<const PatchInfo<D>>>>(local_parents.begin(), local_parents.end());
154  // find local coarse patches that are ghost paches on other ranks
155  std::map<int, std::map<int, int>> ranks_and_local_patches; // map from rank -> (map of ids -> local
156  // indexes). The second map is for when
157  // local indexes are iterated over, they
158  // are sorted by their cooresponding id
159  for (const PatchInfo<D> &pinfo : this->coarser_domain.getPatchInfoVector()) {
160  for (int child_rank : pinfo.child_ranks) {
161  if (child_rank != -1 && child_rank != rank)
162  ranks_and_local_patches[child_rank][pinfo.id] = pinfo.local_index;
163  }
164  }
165 
166  rank_and_local_indexes_for_vector.reserve(ranks_and_local_patches.size());
167  for (auto pair : ranks_and_local_patches) {
168  std::vector<int> local_indexes;
169  // the map sould have sorted patches by id, the other processors will expect things in
170  // this order
171  local_indexes.reserve(pair.second.size());
172  for (auto id_local_index_pair : pair.second) {
173  local_indexes.push_back(id_local_index_pair.second);
174  }
175  rank_and_local_indexes_for_vector.emplace_back(pair.first, local_indexes);
176  }
177 
178  // fill in ghost vector
179  // first, assign local indexes in ghost vector for ghost patches
180  std::map<int, int> id_ghost_vector_local_index_map;
181  int index = 0;
182  for (int id : ghost_parents_ids) {
183  id_ghost_vector_local_index_map[id] = index;
184  index++;
185  }
186 
187  std::map<int, std::map<int, int>> ranks_and_ghost_patches; // map from rank -> (map of ids -> ghost
188  // indexes). The second map is for when
189  // local indexes are iterated over, they
190  // are sorted by their cooresponding id
191  patches_with_ghost_parent.reserve(ghost_parents.size());
192  for (auto patch_ref_wrap : ghost_parents) {
193  const PatchInfo<D> &patch = patch_ref_wrap.get();
194  int ghost_local_index = id_ghost_vector_local_index_map[patch.parent_id];
195 
196  ranks_and_ghost_patches[patch.parent_rank][patch.parent_id] = ghost_local_index;
197 
198  patches_with_ghost_parent.emplace_back(ghost_local_index, patch);
199  }
200 
201  rank_and_local_indexes_for_ghost_vector.reserve(ranks_and_local_patches.size());
202  for (auto pair : ranks_and_ghost_patches) {
203  // the map sould have sorted patches by id, the other processors will expect things in
204  // this order
205  std::vector<int> local_indexes;
206  local_indexes.reserve(pair.second.size());
207  for (auto id_local_index_pair : pair.second) {
208  local_indexes.push_back(id_local_index_pair.second);
209  }
210  rank_and_local_indexes_for_ghost_vector.emplace_back(pair.first, local_indexes);
211  }
212  }

276  void sendGhostPatchesStart(Vector<D> &vector, const Vector<D> &ghost_vector)
277  {
278  if (communicating) {
279  if (sending) {
280  throw RuntimeError("InterLevelComm has a sendGhostPatches posted that is unfinished");
281  } else {
282  throw RuntimeError("InterLevelComm has a getGhostPatches posted that is unfinished");
283  }
284  }
285 
286  // keep track of what vectors we are using
287  current_ghost_vector = &ghost_vector;
288  current_vector = &vector;
289 
290  // post receives
291  recv_buffers.reserve(rank_and_local_indexes_for_vector.size());
292  recv_requests.reserve(rank_and_local_indexes_for_vector.size());
293  for (auto rank_indexes_pair : rank_and_local_indexes_for_vector) {
294  // allocate buffer
295  recv_buffers.emplace_back(vector.getNumComponents() * patch_size * rank_indexes_pair.second.size());
296 
297  // post the receive
298  int rank = rank_indexes_pair.first;
299  recv_requests.emplace_back();
300  MPI_Irecv(recv_buffers.back().data(), recv_buffers.back().size(), MPI_DOUBLE, rank, 0, MPI_COMM_WORLD, &recv_requests.back());
301  }
302  send_buffers.reserve(rank_and_local_indexes_for_ghost_vector.size());
303  send_requests.reserve(rank_and_local_indexes_for_ghost_vector.size());
304  // post sends
305  for (auto rank_indexes_pair : rank_and_local_indexes_for_ghost_vector) {
306  // allocate buffer
307  send_buffers.emplace_back(vector.getNumComponents() * patch_size * rank_indexes_pair.second.size());
308 
309  // fill buffer with values
310  int buffer_idx = 0;
311  for (int local_index : rank_indexes_pair.second) {
312  PatchView<const double, D> view = ghost_vector.getPatchView(local_index);
313  loop_over_all_indexes<D + 1>(view, [&](const std::array<int, D + 1> &coord) {
314  send_buffers.back()[buffer_idx] = view[coord];
315  buffer_idx++;
316  });
317  }
318 
319  // post the send
320  int rank = rank_indexes_pair.first;
321  send_requests.emplace_back();
322  MPI_Isend(send_buffers.back().data(), send_buffers.back().size(), MPI_DOUBLE, rank, 0, MPI_COMM_WORLD, &send_requests.back());
323  }
324 
325  // set state
326  communicating = true;
327  sending = true;
328  }
341  void sendGhostPatchesFinish(Vector<D> &vector, const Vector<D> &ghost_vector)
342  {
343  if (!communicating) {
344  throw RuntimeError("InterLevelComm cannot finish sendGhostPatches since communication was not started");
345  } else if (!sending) {
346  throw RuntimeError("InterLevelComm sendGhostPatchesFinish is being called after getGhostPatchesStart was called");
347  }
348  if (&vector != current_vector) {
349  throw RuntimeError(
350  "InterLevelComm sendGhostPatchesFinish is being called with a different vector than when sendGhostPatchesStart was called");
351  }
352  if (&ghost_vector != current_ghost_vector) {
353  throw RuntimeError(
354  "InterLevelComm senGhostPatchesFinish is being called with a different ghost vector than when sendGhostPatchesStart was called");
355  }
356 
357  // finish recvs
358  for (size_t i = 0; i < rank_and_local_indexes_for_vector.size(); i++) {
359  int finished_idx;
360  MPI_Waitany(recv_requests.size(), recv_requests.data(), &finished_idx, MPI_STATUS_IGNORE);
361 
362  // get local indexes for the buffer that was received
363  const std::vector<int> &local_indexes = rank_and_local_indexes_for_vector.at(finished_idx).second;
364 
365  // add the values in the buffer to the vector
366  std::vector<double> &buffer = recv_buffers.at(finished_idx);
367  int buffer_idx = 0;
368  for (int local_index : local_indexes) {
369  PatchView<double, D> view = vector.getPatchView(local_index);
370  loop_over_all_indexes<D + 1>(view, [&](const std::array<int, D + 1> &coord) {
371  view[coord] += buffer[buffer_idx];
372  buffer_idx++;
373  });
374  }
375  }
376 
377  // wait for sends for finish
378  MPI_Waitall(send_requests.size(), send_requests.data(), MPI_STATUS_IGNORE);
379 
380  // clear buffers
381  recv_requests.clear();
382  recv_buffers.clear();
383  send_requests.clear();
384  send_buffers.clear();
385 
386  // set state
387  communicating = false;
388  current_ghost_vector = nullptr;
389  current_vector = nullptr;
390  }
