379  std::array<Face<D, D - 1>, D - M> getSides() const
380  {
381  std::array<Face<D, D - 1>, D - M> sides;
382  if constexpr (D > 1 && M == 0) {
383  for (size_t i = 0; i < D; i++) {
384  size_t side = 2 * i;
385  if ((1 << i) & value) {
386  side |= 1;
387  }
388  sides[i] = Face<D, D - 1>(side);
389  }
390  } else if constexpr (D == 3 && M == 1) {
391  size_t not_axis = value >> 2;
392  size_t curr_index = 0;
393  for (unsigned char i = 0; i < D; i++) {
394  if (i != not_axis) {
395  unsigned char bit = (value & (0b1 << curr_index)) >> curr_index;
396  sides[curr_index] = Face<D, D - 1>((i << 1) ^ bit);
397  curr_index++;
398  }
399  }
400  } else {
401  sides[0] = Face<D, D - 1>(value);
402  }
403  return sides;
404  }
