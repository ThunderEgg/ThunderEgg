51  PatchArray(const std::array<int, D> &lengths, int num_components, int num_ghost_cells)
52  {
53  std::array<int, D + 1> view_lengths;
54  for (int i = 0; i < D; i++) {
55  view_lengths[i] = lengths[i];
56  }
57  view_lengths[D] = num_components;
58  std::array<int, D + 1> strides;
59  strides[0] = 1;
60  for (int i = 1; i < D + 1; i++) {
61  strides[i] = strides[i - 1] * (lengths[i - 1] + 2 * num_ghost_cells);
62  }
63  int size = strides[D] * num_components;
64  vector.resize(size);
65  view = PatchView<double, D>(vector.data(), strides, view_lengths, num_ghost_cells);
66  }
