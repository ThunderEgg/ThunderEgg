41  protected:
48  void visit(const Level<D> &level, const Vector<D> &f, Vector<D> &u) const override
49  {
50  if (level.coarsest()) {
51  for (int i = 0; i < num_coarse_sweeps; i++) {
52  level.getSmoother().smooth(f, u);
53  }
54  } else {
55  for (int i = 0; i < num_pre_sweeps; i++) {
56  level.getSmoother().smooth(f, u);
57  }
58 
59  Vector<D> coarser_f = this->restrict(level, f, u);
60 
61  const Level<D> &coarser_level = *level.getCoarser();
62  Vector<D> coarser_u = coarser_f.getZeroClone();
63 
64  this->visit(*level.getCoarser(), coarser_f, coarser_u);
65 
66  coarser_level.getInterpolator().interpolate(coarser_u, u);
67 
68  for (int i = 0; i < num_mid_sweeps; i++) {
69  level.getSmoother().smooth(f, u);
70  }
71 
72  coarser_f = this->restrict(level, f, u);
73 
74  this->visit(*level.getCoarser(), coarser_f, coarser_u);
75 
76  coarser_level.getInterpolator().interpolate(coarser_u, u);
77 
78  for (int i = 0; i < num_post_sweeps; i++) {
79  level.getSmoother().smooth(f, u);
80  }
81  }
82  }
83

90  WCycle(std::shared_ptr<Level<D>> finest_level, const CycleOpts &opts) : Cycle<D>(finest_level)
91  {
92  num_pre_sweeps = opts.pre_sweeps;
93  num_post_sweeps = opts.post_sweeps;

95  num_mid_sweeps = opts.mid_sweeps;
96  }
