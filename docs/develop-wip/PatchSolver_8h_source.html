102  virtual void apply(const Vector<D> &f, Vector<D> &u) const override
103  {
104  if constexpr (ENABLE_DEBUG) {
105  if (u.getNumLocalPatches() != this->domain.getNumLocalPatches()) {
106  throw RuntimeError("u vector is incorrect length");
107  }
108  if (f.getNumLocalPatches() != this->domain.getNumLocalPatches()) {

112  u.setWithGhost(0);
113  if (domain.hasTimer()) {
114  domain.getTimer()->startDomainTiming(domain.getId(), "Total Patch Solve");
115  }
116  for (const PatchInfo<D> &pinfo : domain.getPatchInfoVector()) {
117  if (domain.hasTimer()) {
118  domain.getTimer()->start("Single Patch Solve");
119  }
120  PatchView<const double, D> f_view = f.getPatchView(pinfo.local_index);
121  PatchView<double, D> u_view = u.getPatchView(pinfo.local_index);
122  solveSinglePatch(pinfo, f_view, u_view);
123  if (domain.hasTimer()) {
124  domain.getTimer()->stop("Single Patch Solve");
125  }
126  }
127  if (domain.hasTimer()) {
128  domain.getTimer()->stopDomainTiming(domain.getId(), "Total Patch Solve");
129  }
130  }
137  virtual void smooth(const Vector<D> &f, Vector<D> &u) const override
138  {
139  if constexpr (ENABLE_DEBUG) {
140  if (u.getNumLocalPatches() != this->domain.getNumLocalPatches()) {
141  throw RuntimeError("u vector is incorrect length");

144  throw RuntimeError("f vector is incorrect length");
145  }
146  }
147  if (domain.hasTimer()) {
148  domain.getTimer()->startDomainTiming(domain.getId(), "Total Patch Smooth");
149  }
150  ghost_filler->fillGhost(u);
151  for (const PatchInfo<D> &pinfo : domain.getPatchInfoVector()) {
152  if (domain.hasTimer()) {
153  domain.getTimer()->startPatchTiming(pinfo.id, domain.getId(), "Single Patch Solve");
154  }
155  PatchView<const double, D> f_view = f.getPatchView(pinfo.local_index);
156  PatchView<double, D> u_view = u.getPatchView(pinfo.local_index);
157  solveSinglePatch(pinfo, f_view, u_view);
158  if (domain.hasTimer()) {
159  domain.getTimer()->stopPatchTiming(pinfo.id, domain.getId(), "Single Patch Solve");
160  }
161  }
