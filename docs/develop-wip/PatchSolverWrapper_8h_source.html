37 template <int D> class PatchSolverWrapper : public Operator<D - 1>
38 {
39  private:
43  std::shared_ptr<const InterfaceDomain<D>> iface_domain;
47  std::shared_ptr<const PatchSolver<D>> solver;
51  PatchIfaceScatter<D> scatter;
55  std::deque<std::shared_ptr<const PatchIfaceInfo<D>>> patches_with_only_local_ifaces;
59  std::deque<std::shared_ptr<const PatchIfaceInfo<D>>> patches_with_ifaces_on_neighbor_rank;
60 
61  public:
68  PatchSolverWrapper(const InterfaceDomain<D> &iface_domain, const PatchSolver<D> &solver)
69  : iface_domain(std::make_shared<InterfaceDomain<D>>(iface_domain)),
70  solver(solver.clone()),
71  scatter(iface_domain)
72  {
73  int rank;
74  MPI_Comm_rank(MPI_COMM_WORLD, &rank);
75  for (auto piinfo : iface_domain.getPatchIfaceInfos()) {
76  for (Side<D> s : Side<D>::getValues()) {
77  if (piinfo->pinfo.hasNbr(s) && piinfo->getIfaceInfo(s)->rank != rank) {
78  patches_with_ifaces_on_neighbor_rank.push_back(piinfo);
79  break;
80  }
81  }
82  if (patches_with_ifaces_on_neighbor_rank.empty() || patches_with_ifaces_on_neighbor_rank.back() != piinfo) {
83  patches_with_only_local_ifaces.push_back(piinfo);
84  }
85  }
86  }

102  void apply(const Vector<D - 1> &x, Vector<D - 1> &b) const override
103  {
104  int rank;
105  MPI_Comm_rank(MPI_COMM_WORLD, &rank);
106 
107  Vector<D> u(solver->getDomain(), 1);
108  Vector<D> f(solver->getDomain(), 1);
109 
110  // scatter local iface vector
111  auto local_x = scatter.getNewLocalPatchIfaceVector();
112  auto state = scatter.scatterStart(x, *local_x);
113 
114  // each patch has a local interface, go ahead and set the ghost values using those
115  // interfaces
116  for (auto piinfo : iface_domain->getPatchIfaceInfos()) {
117  for (Side<D> s : Side<D>::getValues()) {
118  auto local_data = u.getComponentView(0, piinfo->pinfo.local_index);
119  if (piinfo->pinfo.hasNbr(s) && piinfo->getIfaceInfo(s)->rank == rank) {
120  auto ghosts = local_data.getSliceOn(s, {-1});
121  auto interface = local_x->getComponentView(0, piinfo->getIfaceInfo(s)->patch_local_index);
122  nested_loop<D - 1>(
123  interface.getStart(), interface.getEnd(), [&](const std::array<int, D - 1> &coord) { ghosts[coord] = 2 * interface[coord]; });
124  }
125  }
126  }
127  // go ahead and solve for patches with only local interfaces
128  for (auto piinfo : patches_with_only_local_ifaces) {
129  PatchView<double, D> u_view = u.getPatchView(piinfo->pinfo.local_index);
130  PatchView<double, D> f_view = f.getPatchView(piinfo->pinfo.local_index);
131  solver->solveSinglePatch(piinfo->pinfo, f_view, u_view);
132  }
133 
134  scatter.scatterFinish(state, x, *local_x);
135 
136  // set ghosts using interfaces that were on a neighboring rank
137  for (auto piinfo : patches_with_ifaces_on_neighbor_rank) {
138  for (Side<D> s : Side<D>::getValues()) {
139  auto local_data = u.getComponentView(0, piinfo->pinfo.local_index);
140  if (piinfo->pinfo.hasNbr(s) && piinfo->getIfaceInfo(s)->rank != rank) {
141  auto ghosts = local_data.getSliceOn(s, {-1});
142  auto interface = local_x->getComponentView(0, piinfo->getIfaceInfo(s)->patch_local_index);
143  nested_loop<D - 1>(
144  interface.getStart(), interface.getEnd(), [&](const std::array<int, D - 1> &coord) { ghosts[coord] = 2 * interface[coord]; });
145  }
146  }
147  }
148  // solve the remaining patches
149  for (auto piinfo : patches_with_ifaces_on_neighbor_rank) {
150  PatchView<double, D> u_view = u.getPatchView(piinfo->pinfo.local_index);
151  PatchView<double, D> f_view = f.getPatchView(piinfo->pinfo.local_index);
152  solver->solveSinglePatch(piinfo->pinfo, f_view, u_view);
153  }
154 
155  solver->getGhostFiller().fillGhost(u);
156 
157  for (auto iface : iface_domain->getInterfaces()) {
158  for (auto patch : iface->patches) {
159  if (patch.piinfo->pinfo.rank == rank && (patch.type.isNormal() || patch.type.isCoarseToCoarse() || patch.type.isFineToFine())) {
160  auto local_data = u.getComponentView(0, patch.piinfo->pinfo.local_index);
161  auto ghosts = local_data.getSliceOn(patch.side, {-1});
162  auto inner = local_data.getSliceOn(patch.side, {0});
163  auto interface = b.getComponentView(0, patch.piinfo->getIfaceInfo(patch.side)->patch_local_index);
164  nested_loop<D - 1>(interface.getStart(), interface.getEnd(), [&](const std::array<int, D - 1> &coord) {
165  interface[coord] = (ghosts[coord] + inner[coord]) / 2;
166  });
167  break;
168  }
169  }
170  }
171  b.scaleThenAdd(-1, x);
172  }
179  void getSchurRHSFromDomainRHS(const Vector<D> &domain_b, Vector<D - 1> &schur_b) const
180  {
181  int rank;
182  MPI_Comm_rank(MPI_COMM_WORLD, &rank);
183 
184  Vector<D> u(solver->getDomain(), 1);
185 
186  for (auto piinfo : iface_domain->getPatchIfaceInfos()) {
187  for (Side<D> s : Side<D>::getValues()) {
188  auto local_data = u.getComponentView(0, piinfo->pinfo.local_index);
189  if (piinfo->pinfo.hasNbr(s)) {
190  auto ghosts = local_data.getSliceOn(s, {-1});
191  auto inner = local_data.getSliceOn(s, {0});
192  nested_loop<D - 1>(
193  ghosts.getStart(), ghosts.getEnd(), [&](const std::array<int, D - 1> &coord) { ghosts[coord] = -inner[coord]; });
194  }
195  }
196  }
197  for (auto piinfo : iface_domain->getPatchIfaceInfos()) {
198  PatchView<double, D> u_view = u.getPatchView(piinfo->pinfo.local_index);
199  PatchView<const double, D> f_view = domain_b.getPatchView(piinfo->pinfo.local_index);
200  solver->solveSinglePatch(piinfo->pinfo, f_view, u_view);
201  }
202 
203  solver->getGhostFiller().fillGhost(u);
204 
205  for (auto iface : iface_domain->getInterfaces()) {
206  for (auto patch : iface->patches) {
207  if (patch.piinfo->pinfo.rank == rank && (patch.type.isNormal() || patch.type.isCoarseToCoarse() || patch.type.isFineToFine())) {
208  auto local_data = u.getComponentView(0, patch.piinfo->pinfo.local_index);
209  auto ghosts = local_data.getSliceOn(patch.side, {-1});
210  auto inner = local_data.getSliceOn(patch.side, {0});
211  auto interface = schur_b.getComponentView(0, patch.piinfo->getIfaceInfo(patch.side)->patch_local_index);
212  nested_loop<D - 1>(interface.getStart(), interface.getEnd(), [&](const std::array<int, D - 1> &coord) {
213  interface[coord] = (ghosts[coord] + inner[coord]) / 2;
214  });
215  break;
216  }
217  }
218  }
219  }
