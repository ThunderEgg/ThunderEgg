82  }
83  if (fine.getNumLocalPatches() != ilc->getFinerDomain()->getNumLocalPatches()) {
84  throw RuntimeError("fine vector is incorrect length. Expected Lenght of "
85  + std::to_string(ilc->getFinerDomain()->getNumLocalPatches()) + " but vector was length "
86  + std::to_string(fine.getNumLocalPatches()));
87  }
88  }
89  Vector<D> coarse_ghost = ilc.getNewGhostVector(coarse.getNumComponents());
90 
91  // start scatter for ghost values
92  ilc.getGhostPatchesStart(coarse, coarse_ghost);
93 
94  // interpolate form local values
95  interpolatePatches(ilc.getPatchesWithLocalParent(), coarse, fine);
96 
97  // finish scatter for ghost values
98  ilc.getGhostPatchesFinish(coarse, coarse_ghost);
99 
100  // interpolator from ghost values
101  interpolatePatches(ilc.getPatchesWithGhostParent(), coarse_ghost, fine);
102  }
