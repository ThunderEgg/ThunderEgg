79  bool patchIsNeumannOnSide(const PatchInfo<D> &pinfo, Side<D> s)

90  std::array<fftw_r2r_kind, D> getTransformsForPatch(const PatchInfo<D> &pinfo)
91  {
92  // get transform types for each axis
93  std::array<fftw_r2r_kind, D> transforms;
94  for (size_t axis = 0; axis < D; axis++) {
95  if (patchIsNeumannOnSide(pinfo, LowerSideOnAxis<D>(axis)) && patchIsNeumannOnSide(pinfo, HigherSideOnAxis<D>(axis))) {
96  transforms[D - 1 - axis] = FFTW_REDFT10;
97  } else if (patchIsNeumannOnSide(pinfo, LowerSideOnAxis<D>(axis))) {
98  transforms[D - 1 - axis] = FFTW_REDFT11;
99  } else if (patchIsNeumannOnSide(pinfo, HigherSideOnAxis<D>(axis))) {
100  transforms[D - 1 - axis] = FFTW_RODFT11;
101  } else {
102  transforms[D - 1 - axis] = FFTW_RODFT10;
103  }
104  }
105  return transforms;
106  }
114  std::array<fftw_r2r_kind, D> getInverseTransformsForPatch(const PatchInfo<D> &pinfo)
115  {
116  // get transform types for each axis
117  std::array<fftw_r2r_kind, D> transforms_inv;
118  for (size_t axis = 0; axis < D; axis++) {
119  if (patchIsNeumannOnSide(pinfo, LowerSideOnAxis<D>(axis)) && patchIsNeumannOnSide(pinfo, HigherSideOnAxis<D>(axis))) {
120  transforms_inv[D - 1 - axis] = FFTW_REDFT01;
121  } else if (patchIsNeumannOnSide(pinfo, LowerSideOnAxis<D>(axis))) {
122  transforms_inv[D - 1 - axis] = FFTW_REDFT11;
123  } else if (patchIsNeumannOnSide(pinfo, HigherSideOnAxis<D>(axis))) {
124  transforms_inv[D - 1 - axis] = FFTW_RODFT11;
125  } else {
126  transforms_inv[D - 1 - axis] = FFTW_RODFT01;
127  }
128  }

137  PatchArray<D> getEigenValues(const PatchInfo<D> &pinfo)
138  {
139  PatchArray<D> retval(this->getDomain().getNs(), 1, 0);
140 
141  std::valarray<size_t> all_strides(D);
142  size_t curr_stride = 1;
143  for (size_t i = 0; i < D; i++) {
144  all_strides[i] = curr_stride;
145  curr_stride *= pinfo.ns[i];
146  }
147 
148  for (size_t axis = 0; axis < D; axis++) {
149  int n = pinfo.ns[axis];
150  double h = pinfo.spacings[axis];
151 
152  if (patchIsNeumannOnSide(pinfo, LowerSideOnAxis<D>(axis)) && patchIsNeumannOnSide(pinfo, HigherSideOnAxis<D>(axis))) {
153  for (int xi = 0; xi < n; xi++) {
154  double val = 4 / (h * h) * pow(sin(xi * M_PI / (2 * n)), 2);
155  View<double, D> slice = retval.getSliceOn(Side<D>(2 * axis), {xi});
156  loop_over_interior_indexes<D>(slice, [&](const std::array<int, D> &coord) { slice[coord] -= val; });
157  }
158  } else if (patchIsNeumannOnSide(pinfo, LowerSideOnAxis<D>(axis)) || patchIsNeumannOnSide(pinfo, HigherSideOnAxis<D>(axis))) {
159  for (int xi = 0; xi < n; xi++) {
160  double val = 4 / (h * h) * pow(sin((xi + 0.5) * M_PI / (2 * n)), 2);
161  View<double, D> slice = retval.getSliceOn(Side<D>(2 * axis), {xi});
162  loop_over_interior_indexes<D>(slice, [&](const std::array<int, D> &coord) { slice[coord] -= val; });
163  }
164  } else {
165  for (int xi = 0; xi < n; xi++) {
166  double val = 4 / (h * h) * pow(sin((xi + 1) * M_PI / (2 * n)), 2);
167  View<double, D> slice = retval.getSliceOn(Side<D>(2 * axis), {xi});
168  loop_over_interior_indexes<D>(slice, [&](const std::array<int, D> &coord) { slice[coord] -= val; });
169  }
170  }
171  }
172  return retval;
173  }

182  FFTWPatchSolver(const PatchOperator<D> &op, std::bitset<Side<D>::number_of> neumann)
183  : PatchSolver<D>(op.getDomain(), op.getGhostFiller()),
184  op(op.clone()),
185  neumann(neumann)
186  {
187  CompareFunction compare = [&](const PatchInfo<D> &a, const PatchInfo<D> &b) {
188  std::bitset<Side<D>::number_of> a_neumann;
189  std::bitset<Side<D>::number_of> b_neumann;
190  for (Side<D> s : Side<D>::getValues()) {
191  a_neumann[s.getIndex()] = patchIsNeumannOnSide(a, s);
192  b_neumann[s.getIndex()] = patchIsNeumannOnSide(b, s);
193  }
194  return std::forward_as_tuple(a_neumann.to_ulong(), a.spacings[0]) < std::forward_as_tuple(b_neumann.to_ulong(), b.spacings[0]);
195  };

215  void solveSinglePatch(const PatchInfo<D> &pinfo, const PatchView<const double, D> &f_view, const PatchView<double, D> &u_view) const override
216  {
217  PatchArray<D> f_copy(pinfo.ns, 1, 0);
218  PatchArray<D> tmp(pinfo.ns, 1, 0);
219  PatchArray<D> sol(pinfo.ns, 1, 0);
220 
221  loop_over_interior_indexes<D + 1>(f_copy, [&](std::array<int, D + 1> coord) { f_copy[coord] = f_view[coord]; });
222 
223  op->modifyRHSForInternalBoundaryConditions(pinfo, u_view, f_copy.getView());
224 
225  fftw_execute_r2r(*plan1.at(pinfo), &f_copy[f_copy.getStart()], &tmp[tmp.getStart()]);
226 
227  const PatchArray<D> &eigen_vals_view = eigen_vals.at(pinfo);
228  loop_over_interior_indexes<D + 1>(tmp, [&](std::array<int, D + 1> coord) { tmp[coord] /= eigen_vals_view[coord]; });
229 
230  if (neumann.all() && !pinfo.hasNbr()) {
231  tmp[tmp.getStart()] = 0;
232  }
233 
234  fftw_execute_r2r(*plan2.at(pinfo), &tmp[tmp.getStart()], &sol[sol.getStart()]);
235 
236  double scale = 1;
237  for (size_t axis = 0; axis < D; axis++) {
238  scale *= 2.0 * this->getDomain().getNs()[axis];
239  }
240  loop_over_interior_indexes<D + 1>(u_view, [&](std::array<int, D + 1> coord) { u_view[coord] = sol[coord] / scale; });
241  }
249  void addPatch(const PatchInfo<D> &pinfo)
250  {
251  if (plan1.count(pinfo) == 0) {
252  // revers ns because FFTW is row major
253  std::array<int, D> ns_reversed;
254  for (size_t i = 0; i < D; i++) {
255  ns_reversed[D - 1 - i] = pinfo.ns[i];
256  }
257  std::array<fftw_r2r_kind, D> transforms = getTransformsForPatch(pinfo);
258  std::array<fftw_r2r_kind, D> transforms_inv = getInverseTransformsForPatch(pinfo);
259 
260  PatchArray<D> f_copy(pinfo.ns, 1, 0);
261  PatchArray<D> tmp(pinfo.ns, 1, 0);
262  PatchArray<D> sol(pinfo.ns, 1, 0);
263 
264  fftw_plan *fftw_plan1 = new fftw_plan();
265 
266  *fftw_plan1 = fftw_plan_r2r(D,
267  ns_reversed.data(),
268  &f_copy[f_copy.getStart()],
269  &tmp[tmp.getStart()],
270  transforms.data(),
271  FFTW_MEASURE | FFTW_DESTROY_INPUT | FFTW_UNALIGNED);
272 
273  plan1[pinfo] = std::shared_ptr<fftw_plan>(fftw_plan1, [](fftw_plan *plan) {
274  fftw_destroy_plan(*plan);
275  delete plan;
276  });
277 
278  fftw_plan *fftw_plan2 = new fftw_plan();
279 
280  *fftw_plan2 = fftw_plan_r2r(D,
281  ns_reversed.data(),
282  &tmp[tmp.getStart()],
283  &sol[sol.getStart()],
284  transforms_inv.data(),
285  FFTW_MEASURE | FFTW_DESTROY_INPUT | FFTW_UNALIGNED);
286 
287  plan2[pinfo] = std::shared_ptr<fftw_plan>(fftw_plan2, [](fftw_plan *plan) {
288  fftw_destroy_plan(*plan);
289  delete plan;
290  });
291 
292  eigen_vals.emplace(pinfo, getEigenValues(pinfo));
293  }
294  }

302  return neumann;
303  }
