126  void insert(Side<D> s, std::shared_ptr<const PatchIfaceInfo<D>> piinfo)
127  {
128  NbrType nbr_type = piinfo->pinfo.getNbrType(s);
129  if (nbr_type == NbrType::Normal) {
130  patches.emplace_back(s, IfaceType<D>::Normal(), piinfo);
131  } else if (nbr_type == NbrType::Fine) {
132  auto info = piinfo->getFineIfaceInfo(s);
133  if (info->id == id) {
134  patches.emplace_back(s, IfaceType<D>::CoarseToCoarse(), piinfo);
135  } else {
136  for (Orthant<D - 1> o : Orthant<D - 1>::getValues()) {
137  if (info->fine_ids[o.getIndex()] == id) {
138  patches.emplace_back(s, IfaceType<D>::CoarseToFine(o), piinfo);
139  break;
140  }
141  }
142  }
143  } else if (nbr_type == NbrType::Coarse) {
144  auto info = piinfo->getCoarseIfaceInfo(s);
145  if (info->id == id) {
146  patches.emplace_back(s, IfaceType<D>::FineToFine(info->orth_on_coarse), piinfo);
147  } else {
148  patches.emplace_back(s, IfaceType<D>::FineToCoarse(info->orth_on_coarse), piinfo);
149  }
150  } else {
151  throw RuntimeError("Unsupported NbrType");
152  }
153  }
159  void merge(Interface<D> ifs)

161  for (auto patch : ifs.patches) {
162  patches.push_back(patch);

304  static void EnumerateIfacesFromPiinfoVector(std::vector<std::shared_ptr<const PatchIfaceInfo<D>>> piinfos,
305  std::map<int, std::map<int, std::shared_ptr<Interface<D>>>> &rank_id_iface_map,
306  std::vector<std::shared_ptr<PatchIfaceInfo<D>>> & off_proc_piinfos)
307  {
308  int rank;
309  MPI_Comm_rank(MPI_COMM_WORLD, &rank);
310  rank_id_iface_map.clear();
311  std::set<int> incoming_procs;
312  for (auto piinfo : piinfos) {
313  for (Side<D> s : Side<D>::getValues()) {
314  if (piinfo->pinfo.hasNbr(s)) {
315  switch (piinfo->pinfo.getNbrType(s)) {
316  case NbrType::Normal:
317  InsertInterfaceWithNormalNbr(rank_id_iface_map, incoming_procs, piinfo, s);
318  break;
319  case NbrType::Fine:
320  InsertInterfacesWithFineNbr(rank_id_iface_map, incoming_procs, piinfo, s);
321  break;
322  case NbrType::Coarse:
323  InsertInterfacesWithCoarseNbr(rank_id_iface_map, incoming_procs, piinfo, s);
324  break;
325  default:
326  throw RuntimeError("Unsupported NbrType value");
327  }
328  }
329  }
330  }
331  // send info
332  std::deque<std::vector<char>> buffers;
333  std::vector<MPI_Request> send_requests;
334  for (auto &p : rank_id_iface_map) {
335  int dest = p.first;
336  if (dest != rank) {
337  int size = 0;
338  for (auto q : p.second) {
339  auto &iface = q.second;
340  size += iface->serialize(nullptr);
341  }
342  buffers.emplace_back(size);
343  BufferWriter writer(buffers.back().data());
344  for (auto q : p.second) {
345  auto &iface = q.second;
346  writer << *iface;
347  }
348  MPI_Request request;
349  MPI_Isend(buffers.back().data(), size, MPI_BYTE, dest, 0, MPI_COMM_WORLD, &request);
350  send_requests.push_back(request);
351  }
352  }
353  // recv info
354  std::map<int, std::shared_ptr<PatchIfaceInfo<D>>> id_to_off_proc_piinfo_map;
355 
356  MPI_Barrier(MPI_COMM_WORLD);
357 
358  size_t num_incoming = incoming_procs.size();
359  for (size_t i = 0; i < num_incoming; i++) {
360  MPI_Status status;
361  MPI_Probe(MPI_ANY_SOURCE, 0, MPI_COMM_WORLD, &status);
362  int size;
363  MPI_Get_count(&status, MPI_BYTE, &size);
364  std::vector<char> buffer(size);
365 
366  MPI_Recv(buffer.data(), size, MPI_BYTE, status.MPI_SOURCE, 0, MPI_COMM_WORLD, MPI_STATUS_IGNORE);
367 
368  BufferReader reader(buffer.data());
369  while (reader.getPos() < size) {
370  Interface<D> ifs;
371  reader >> ifs;
372  for (auto &patch : ifs.patches) {
373  auto &ptr = id_to_off_proc_piinfo_map[patch.piinfo->pinfo.id];
374  if (ptr == nullptr) {
375  // need to cast to remove const modifier
376  ptr = std::const_pointer_cast<PatchIfaceInfo<D>>(patch.piinfo);
377  } else {
378  patch.piinfo = ptr;
379  }
380  }
381  rank_id_iface_map.at(rank).at(ifs.id)->merge(ifs);
382  }
383  off_proc_piinfos.clear();
384  for (auto pair : id_to_off_proc_piinfo_map) {
385  off_proc_piinfos.push_back(pair.second);
386  }
387  }
388  // wait for all
389  MPI_Waitall((int) send_requests.size(), &send_requests[0], MPI_STATUSES_IGNORE);
390  }
