117  std::shared_ptr<std::valarray<double>> getTransformArray(DftType type, int n)
118  {
119  std::shared_ptr<std::valarray<double>> matrix_ptr;
120 
121  if (transform_matrixes.count(std::make_tuple(type, n)) == 0) {
122  matrix_ptr = std::make_shared<std::valarray<double>>(n * n);
123  transform_matrixes[std::make_tuple(type, n)] = matrix_ptr;
124  std::valarray<double> &matrix = *matrix_ptr;
125 
126  switch (type) {
127  case DftType::DCT_II:
128  for (int j = 0; j < n; j++) {
129  for (int i = 0; i < n; i++) {
130  matrix[i * n + j] = cos(M_PI / n * (i * (j + 0.5)));
131  }
132  }
133  break;
134  case DftType::DCT_III:
135  for (int i = 0; i < n; i++) {
136  matrix[i * n] = 0.5;
137  }
138  for (int j = 1; j < n; j++) {
139  for (int i = 0; i < n; i++) {
140  matrix[i * n + j] = cos(M_PI / n * ((i + 0.5) * j));
141  }
142  }
143  break;
144  case DftType::DCT_IV:
145  for (int j = 0; j < n; j++) {
146  for (int i = 0; i < n; i++) {
147  matrix[i * n + j] = cos(M_PI / n * ((i + 0.5) * (j + 0.5)));
148  }
149  }
150  break;
151  case DftType::DST_II:
152  for (int i = 0; i < n; i++) {
153  for (int j = 0; j < n; j++) {
154  matrix[i * n + j] = sin(M_PI / n * ((i + 1) * (j + 0.5)));
155  }
156  }
157  break;
158  case DftType::DST_III:
159  for (int i = 0; i < n; i += 2) {
160  matrix[i * n + n - 1] = 0.5;
161  }
162  for (int i = 1; i < n; i += 2) {
163  matrix[i * n + n - 1] = -0.5;
164  }
165  for (int i = 0; i < n; i++) {
166  for (int j = 0; j < n - 1; j++) {
167  matrix[i * n + j] = sin(M_PI / n * ((i + 0.5) * (j + 1)));
168  }
169  }
170  break;
171  case DftType::DST_IV:
172  for (int j = 0; j < n; j++) {
173  for (int i = 0; i < n; i++) {
174  matrix[i * n + j] = sin(M_PI / n * ((i + 0.5) * (j + 0.5)));
175  }
176  }
177  }
178  } else {
179  matrix_ptr = transform_matrixes.at(std::make_tuple(type, n));
180  }
181  return matrix_ptr;
182  }
191  void executePlan(const std::array<std::shared_ptr<std::valarray<double>>, D> &plan,
192  const PatchView<double, D> & in,
193  const PatchView<double, D> & out) const
194  {
195  PatchView<double, D> prev_result = in;
196 
197  PatchArray<D> local_tmp;
198  if (!(D % 2)) {
199  local_tmp = PatchArray<D>(op->getDomain().getNs(), 1, 0);
200  }
201 
202  for (size_t axis = 0; axis < D; axis++) {
203  int n = this->getDomain().getNs()[axis];
204  std::array<int, D + 1> start = in.getStart();
205  std::array<int, D + 1> end = in.getEnd();
206  end[axis] = 0;
207 
208  std::valarray<double> &matrix = *plan[axis];
209 
210  PatchView<double, D> new_result;
211  if (D % 2) {
212  if (axis % 2) {
213  new_result = in;
214  } else {
215  new_result = out;
216  }
217  } else {
218  if (axis == D - 1) {
219  new_result = out;
220  } else if (axis == D - 2) {
221  new_result = local_tmp.getView();
222  } else if (axis % 2) {
223  new_result = in;
224  } else {
225  new_result = out;
226  }
227  }
228 
229  int pstride = prev_result.getStrides()[axis];
230  int nstride = new_result.getStrides()[axis];
231 
232  char T = 'T';
233  double one = 1;
234  double zero = 0;
235  nested_loop<D + 1>(start, end, [&](std::array<int, D + 1> coord) {
236  dgemv_(T, n, n, one, &matrix[0], n, &prev_result[coord], pstride, zero, &new_result[coord], nstride);
237  });
238 
239  prev_result = new_result;
240  }
241  }
248  std::array<DftType, D> getTransformsForPatch(const ThunderEgg::PatchInfo<D> &pinfo)
249  {
250  // get transform types for each axis
251  std::array<DftType, D> transforms;
252  for (size_t axis = 0; axis < D; axis++) {
253  if (patchIsNeumannOnSide(pinfo, LowerSideOnAxis<D>(axis)) && patchIsNeumannOnSide(pinfo, HigherSideOnAxis<D>(axis))) {
254  transforms[axis] = DftType::DCT_II;
255  } else if (patchIsNeumannOnSide(pinfo, LowerSideOnAxis<D>(axis))) {
256  transforms[axis] = DftType::DCT_IV;
257  } else if (patchIsNeumannOnSide(pinfo, HigherSideOnAxis<D>(axis))) {
258  transforms[axis] = DftType::DST_IV;
259  } else {
260  transforms[axis] = DftType::DST_II;
261  }
262  }
263  return transforms;
264  }

294  PatchArray<D> getEigenValues(const PatchInfo<D> &pinfo)
295  {
296  PatchArray<D> retval(this->getDomain().getNs(), 1, 0);
297 
298  std::valarray<size_t> all_strides(D);
299  size_t curr_stride = 1;
300  for (size_t i = 0; i < D; i++) {
301  all_strides[i] = curr_stride;
302  curr_stride *= pinfo.ns[i];
303  }
304 
305  for (size_t axis = 0; axis < D; axis++) {
306  int n = pinfo.ns[axis];
307  double h = pinfo.spacings[axis];
308 
309  if (patchIsNeumannOnSide(pinfo, LowerSideOnAxis<D>(axis)) && patchIsNeumannOnSide(pinfo, HigherSideOnAxis<D>(axis))) {
310  for (int xi = 0; xi < n; xi++) {
311  double val = 4 / (h * h) * pow(sin(xi * M_PI / (2 * n)), 2);
312  View<double, D> slice = retval.getSliceOn(Side<D>(2 * axis), {xi});
313  loop_over_interior_indexes<D>(slice, [&](const std::array<int, D> &coord) { slice[coord] -= val; });
314  }
315  } else if (patchIsNeumannOnSide(pinfo, LowerSideOnAxis<D>(axis)) || patchIsNeumannOnSide(pinfo, HigherSideOnAxis<D>(axis))) {
316  for (int xi = 0; xi < n; xi++) {
317  double val = 4 / (h * h) * pow(sin((xi + 0.5) * M_PI / (2 * n)), 2);
318  View<double, D> slice = retval.getSliceOn(Side<D>(2 * axis), {xi});
319  loop_over_interior_indexes<D>(slice, [&](const std::array<int, D> &coord) { slice[coord] -= val; });
320  }
321  } else {
322  for (int xi = 0; xi < n; xi++) {
323  double val = 4 / (h * h) * pow(sin((xi + 1) * M_PI / (2 * n)), 2);
324  View<double, D> slice = retval.getSliceOn(Side<D>(2 * axis), {xi});
325  loop_over_interior_indexes<D>(slice, [&](const std::array<int, D> &coord) { slice[coord] -= val; });
326  }
327  }
328  }
329  return retval;
330  }
338  void addPatch(const PatchInfo<D> &pinfo)
339  {
340  if (plan1.count(pinfo) == 0) {
341  plan1[pinfo] = plan(getTransformsForPatch(pinfo));
342  plan2[pinfo] = plan(getInverseTransformsForPatch(pinfo));
343  eigen_vals.emplace(pinfo, getEigenValues(pinfo));
344  }

354  DFTPatchSolver(const PatchOperator<D> &op, std::bitset<Side<D>::number_of> neumann)
355  : PatchSolver<D>(op.getDomain(), op.getGhostFiller()),
356  op(op.clone()),
357  neumann(neumann)
358  {
359  CompareFunction compare = [&](const PatchInfo<D> &a, const PatchInfo<D> &b) {
360  std::bitset<Side<D>::number_of> a_neumann;
361  std::bitset<Side<D>::number_of> b_neumann;
362  for (Side<D> s : Side<D>::getValues()) {
363  a_neumann[s.getIndex()] = patchIsNeumannOnSide(a, s);
364  b_neumann[s.getIndex()] = patchIsNeumannOnSide(b, s);
365  }
366  return std::forward_as_tuple(a_neumann.to_ulong(), a.spacings[0]) < std::forward_as_tuple(b_neumann.to_ulong(), b.spacings[0]);
367  };
368 
369  plan1 = std::map<PatchInfo<D>, std::array<std::shared_ptr<std::valarray<double>>, D>, CompareFunction>(compare);
370  plan2 = std::map<PatchInfo<D>, std::array<std::shared_ptr<std::valarray<double>>, D>, CompareFunction>(compare);
371  eigen_vals = std::map<const PatchInfo<D>, PatchArray<D>, CompareFunction>(compare);
372 
373  // process patches
374  for (auto pinfo : this->getDomain().getPatchInfoVector()) {
375  addPatch(pinfo);
376  }

387  void solveSinglePatch(const PatchInfo<D> &pinfo, const PatchView<const double, D> &f_view, const PatchView<double, D> &u_view) const override
388  {
389  PatchArray<D> f_copy(op->getDomain().getNs(), 1, 0);
390  PatchArray<D> tmp(op->getDomain().getNs(), 1, 0);
391 
392  loop_over_interior_indexes<D + 1>(f_view, [&](std::array<int, D + 1> coord) { f_copy[coord] = f_view[coord]; });
393 
394  op->modifyRHSForInternalBoundaryConditions(pinfo, u_view, f_copy.getView());
395 
396  executePlan(plan1.at(pinfo), f_copy.getView(), tmp.getView());
397 
398  const PatchArray<D> &eigen_vals_view = eigen_vals.at(pinfo);
399  loop_over_interior_indexes<D + 1>(tmp, [&](std::array<int, D + 1> coord) { tmp[coord] /= eigen_vals_view[coord]; });
400 
401  if (neumann.all() && !pinfo.hasNbr()) {
402  tmp[tmp.getStart()] = 0;
403  }
404 
405  executePlan(plan2.at(pinfo), tmp.getView(), u_view);
406 
407  double scale = 1;
408  for (size_t axis = 0; axis < D; axis++) {
409  scale *= 2.0 / this->getDomain().getNs()[axis];
410  }
411  loop_over_interior_indexes<D + 1>(u_view, [&](std::array<int, D + 1> coord) { u_view[coord] *= scale; });
412  }
