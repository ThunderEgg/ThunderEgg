51  Vec getPetscVecWithoutGhost(const Vector<D> &vec) const
52  {
53  Vec petsc_vec;
54  VecCreateMPI(vec.getCommunicator().getMPIComm(), vec.getNumLocalCells() * vec.getNumComponents(), PETSC_DETERMINE, &petsc_vec);

63  void copyToPetscVec(const Vector<D> &vec, Vec petsc_vec) const
64  {
65  double *petsc_vec_view;
66  size_t curr_index = 0;
67  VecGetArray(petsc_vec, &petsc_vec_view);
68  for (int i = 0; i < vec.getNumLocalPatches(); i++) {
69  for (int c = 0; c < vec.getNumComponents(); c++) {
70  const ComponentView<const double, D> ld = vec.getComponentView(c, i);
71  nested_loop<D>(ld.getStart(), ld.getEnd(), [&](const std::array<int, D> &coord) {
72  petsc_vec_view[curr_index] = ld[coord];
73  curr_index++;
74  });
75  }
76  }
77  VecRestoreArray(petsc_vec, &petsc_vec_view);
78  }
85  void copyToVec(Vec petsc_vec, Vector<D> &vec) const
86  {
87  const double *petsc_vec_view;
88  size_t curr_index = 0;
89  VecGetArrayRead(petsc_vec, &petsc_vec_view);
90  for (int i = 0; i < vec.getNumLocalPatches(); i++) {
91  for (int c = 0; c < vec.getNumComponents(); c++) {
92  ComponentView<double, D> ld = vec.getComponentView(c, i);
93  nested_loop<D>(ld.getStart(), ld.getEnd(), [&](const std::array<int, D> &coord) {
94  ld[coord] = petsc_vec_view[curr_index];
95  curr_index++;
96  });
97  }
98  }
99  VecRestoreArrayRead(petsc_vec, &petsc_vec_view);
100  }

141  const Operator<D> *Mr = nullptr,
142  bool output = false,
143  std::ostream & os = std::cout) const override
144  {
145  Mat A_PETSC = MatShellCreator<D>::GetNewMatShell(A, [&]() { return x.getZeroClone(); });
146 
147  PC Mr_PETSC = nullptr;
148  if (Mr != nullptr) {
149  Mr_PETSC = PCShellCreator<D>::GetNewPCShell(*Mr, A, [&]() { return x.getZeroClone(); });
150  }
151  Vec x_PETSC = getPetscVecWithoutGhost(x);
152  copyToPetscVec(x, x_PETSC);
153 
154  Vec b_PETSC = getPetscVecWithoutGhost(x);
155  copyToPetscVec(b, b_PETSC);
156 
157  KSP ksp;
158  KSPCreate(PETSC_COMM_WORLD, &ksp);
159  KSPSetType(ksp, type);
160  KSPSetOperators(ksp, A_PETSC, A_PETSC);
161 
162  if (Mr != nullptr) {
163  KSPSetPC(ksp, Mr_PETSC);
164  // KSPSetPCSide(ksp, PC_RIGHT);
165  }
166 
167  if (output) {
168  KSPMonitorSet(ksp, (PetscErrorCode(*)(KSP, PetscInt, PetscReal, void *)) & MonitorResidual, &os, nullptr);
169  }
170 
171  KSPSolve(ksp, b_PETSC, x_PETSC);
172 
173  int iterations;
174  KSPGetIterationNumber(ksp, &iterations);
175 
176  const char *converged_reason;
177  KSPGetConvergedReasonString(ksp, &converged_reason);
178  os << converged_reason << std::endl;
179 
180  KSPDestroy(&ksp);
181  VecDestroy(&x_PETSC);
182  VecDestroy(&b_PETSC);
183  PCDestroy(&Mr_PETSC);
184  MatDestroy(&A_PETSC);
185  return iterations;
186  }
