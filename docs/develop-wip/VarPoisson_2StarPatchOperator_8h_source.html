80  applySinglePatch(const PatchInfo<D> &pinfo, const PatchView<const double, D> &u_view, const PatchView<double, D> &f_view, bool interior) const
81  {
82  if (interior) {
83  enforceInternalBoundaryConditions(pinfo, u_view);
84  }
85  enforceBoundaryConditions(pinfo, u_view);
86 
87  PatchView<const double, D> c = coeffs.getPatchView(pinfo.local_index);
88  std::array<double, D> h2 = pinfo.spacings;
89  for (size_t i = 0; i < D; i++) {
90  h2[i] *= h2[i];
91  }
92  loop<0, D - 1>([&](int axis) {
93  int stride = u_view.getStrides()[axis];
94  int c_stride = c.getStrides()[axis];
95  loop_over_interior_indexes<D + 1>(u_view, [&](std::array<int, D + 1> coord) {
96  const double *ptr = &u_view[coord];
97  const double *c_ptr = &c[coord];
98  double lower = *(ptr - stride);
99  double mid = *ptr;
100  double upper = *(ptr + stride);
101  double c_lower = *(c_ptr - c_stride);
102  double c_mid = *c_ptr;
103  double c_upper = *(c_ptr + c_stride);
104  f_view[coord]
105  = addValue(axis) * f_view[coord] + ((c_upper + c_mid) * (upper - mid) - (c_lower + c_mid) * (mid - lower)) / (2 * h2[axis]);
106  });
107  });
108  }

121  void enforceBoundaryConditions(const PatchInfo<D> &pinfo, const PatchView<const double, D> &u_view) const
122  {
123  for (int axis = 0; axis < D; axis++) {
124  Side<D> lower_side(axis * 2);
125  Side<D> upper_side(axis * 2 + 1);
126  if (!pinfo.hasNbr(lower_side)) {
127  View<double, D> lower = u_view.getGhostSliceOn(lower_side, {0});
128  View<const double, D> mid = u_view.getSliceOn(lower_side, {0});
129  loop_over_interior_indexes<D>(mid, [&](std::array<int, D> coord) { lower[coord] = -mid[coord]; });
130  }
131  if (!pinfo.hasNbr(upper_side)) {
132  View<double, D> upper = u_view.getGhostSliceOn(upper_side, {0});
133  View<const double, D> mid = u_view.getSliceOn(upper_side, {0});
134  loop_over_interior_indexes<D>(mid, [&](std::array<int, D> coord) { upper[coord] = -mid[coord]; });
135  }
136  }
137  }

139  void enforceInternalBoundaryConditions(const PatchInfo<D> &pinfo, const PatchView<const double, D> &u_view) const
140  {
141  for (int axis = 0; axis < D; axis++) {
142  Side<D> lower_side(axis * 2);
143  Side<D> upper_side(axis * 2 + 1);
144  if (pinfo.hasNbr(lower_side)) {
145  View<double, D> lower = u_view.getGhostSliceOn(lower_side, {0});
146  View<const double, D> mid = u_view.getSliceOn(lower_side, {0});
147  loop_over_interior_indexes<D>(mid, [&](std::array<int, D> coord) { lower[coord] = -mid[coord]; });
148  }
149  if (pinfo.hasNbr(upper_side)) {
150  View<double, D> upper = u_view.getGhostSliceOn(upper_side, {0});
151  View<const double, D> mid = u_view.getSliceOn(upper_side, {0});
152  loop_over_interior_indexes<D>(mid, [&](std::array<int, D> coord) { upper[coord] = -mid[coord]; });
153  }
154  }
155  }

157  void modifyRHSForInternalBoundaryConditions(const PatchInfo<D> & pinfo,
158  const PatchView<const double, D> &u_view,
159  const PatchView<double, D> & f_view) const override
160  {
161  PatchView<const double, D> c = coeffs.getPatchView(pinfo.local_index);
162  for (Side<D> s : Side<D>::getValues()) {
163  if (pinfo.hasNbr(s)) {
164  double h2 = pow(pinfo.spacings[s.getAxisIndex()], 2);
165  View<double, D> fner = f_view.getSliceOn(s, {0});
166  View<double, D> u_ghost = u_view.getGhostSliceOn(s, {0});
167  View<const double, D> uner = u_view.getSliceOn(s, {0});
168  View<const double, D> c_ghost = c.getSliceOn(s, {-1});
169  View<const double, D> cner = c.getSliceOn(s, {0});
170  loop_over_interior_indexes<D>(fner, [&](const std::array<int, D> &coord) {
171  fner[coord] -= (u_ghost[coord] + uner[coord]) * (cner[coord] + c_ghost[coord]) / (2 * h2);
172  });
173  }
174  }
175  }
183  void addDrichletBCToRHS(Vector<D> & f,
184  std::function<double(const std::array<double, D> &)> gfunc,
185  std::function<double(const std::array<double, D> &)> hfunc)
186  {
187  for (int i = 0; i < f.getNumLocalPatches(); i++) {
188  ComponentView<double, D> f_ld = f.getComponentView(0, i);
189  auto pinfo = this->getDomain().getPatchInfoVector()[i];
190  for (Side<D> s : Side<D>::getValues()) {
191  if (!pinfo.hasNbr(s)) {
192  double h2 = pow(pinfo.spacings[s.getAxisIndex()], 2);
193  View<double, D - 1> ld = f_ld.getSliceOn(s, {0});
194  nested_loop<D - 1>(ld.getStart(), ld.getEnd(), [&](const std::array<int, D - 1> &coord) {
195  std::array<double, D> real_coord;
196  DomainTools::GetRealCoordBound<D>(pinfo, coord, s, real_coord);
197  std::array<double, D> other_real_coord = real_coord;
198  if (s.isLowerOnAxis()) {
199  other_real_coord[s.getAxisIndex()] -= pinfo.spacings[s.getAxisIndex()];
200  } else {
201  other_real_coord[s.getAxisIndex()] += pinfo.spacings[s.getAxisIndex()];
202  }
203  ld[coord] -= 2 * gfunc(real_coord) * hfunc(real_coord) / h2;
204  });
205  }
206  }
207  }
208  }
