90  void indexPatchesLocal()
91  {
92  // index patches
93  int curr_index = 0;
94  std::map<int, int> id_to_local_index;
95  for (auto &pinfo : pinfos) {
96  pinfo.local_index = curr_index;
97  id_to_local_index[pinfo.id] = pinfo.local_index;
98  curr_index++;
99  }

109  void indexPatchesGlobal()
110  {
111  // get starting global index
112  int num_local_patches = (int) pinfos.size();
113  int curr_global_index;
114  MPI_Scan(&num_local_patches, &curr_global_index, 1, MPI_INT, MPI_SUM, comm.getMPIComm());
115  curr_global_index -= num_local_patches;
116 
117  // index the patches
118  std::map<int, int> id_to_global_index;
119  for (auto &pinfo : pinfos) {
120  pinfo.global_index = curr_global_index;
121  id_to_global_index[pinfo.id] = pinfo.global_index;
122  curr_global_index++;
123  }
124 
125  std::map<int, std::set<std::pair<int, int>>> ranks_to_ids_and_global_indexes_outgoing;
126  std::map<int, std::set<int>> ranks_to_ids_incoming;
127  for (auto &pinfo : pinfos) {
128  auto ranks = pinfo.getNbrRanks();
129  auto ids = pinfo.getNbrIds();
130  for (size_t idx = 0; idx < ranks.size(); idx++) {
131  int nbr_id = ids[idx];
132  int nbr_rank = ranks[idx];
133  if (nbr_rank != comm.getRank()) {
134  ranks_to_ids_and_global_indexes_outgoing[nbr_rank].insert(std::make_pair(pinfo.id, pinfo.global_index));
135  ranks_to_ids_incoming[nbr_rank].insert(nbr_id);
136  }
137  }
138  }
139 
140  // prepare to recieve data
141  std::vector<MPI_Request> recv_requests;
142 
143  // allocate incoming vectors and post recvs
144  std::map<int, std::vector<int>> rank_to_incoming_data;
145  for (const auto &pair : ranks_to_ids_incoming) {
146  int source_rank = pair.first;
147  std::vector<int> &incoming_data = rank_to_incoming_data[source_rank];
148  incoming_data.resize(pair.second.size());
149 
150  MPI_Request request;
151  MPI_Irecv(incoming_data.data(), (int) incoming_data.size(), MPI_INT, source_rank, 0, comm.getMPIComm(), &request);
152  recv_requests.push_back(request);
153  }
154 
155  // prepare outgoing vector of data and send it
156  std::vector<MPI_Request> send_requests;
157 
158  // post sends
159  std::map<int, std::vector<int>> rank_to_outgoing_data;
160  for (const auto &pair : ranks_to_ids_and_global_indexes_outgoing) {
161  int dest_rank = pair.first;
162  // allocate and fill vector
163  std::vector<int> &data = rank_to_outgoing_data[dest_rank];
164  data.reserve(pair.second.size());
165  for (const auto &id_and_global_index : pair.second) {
166  data.push_back(id_and_global_index.second);
167  }
168  MPI_Request request;
169  MPI_Isend(data.data(), (int) data.size(), MPI_INT, dest_rank, 0, comm.getMPIComm(), &request);
170  send_requests.push_back(request);
171  }
172 
173  // add global indexes to map as recvs come in
174  for (size_t i = 0; i < recv_requests.size(); i++) {
175  MPI_Status status;
176  int request_index;
177  MPI_Waitany((int) recv_requests.size(), recv_requests.data(), &request_index, &status);
178 
179  int source_rank = status.MPI_SOURCE;
180  const std::set<int> & incoming_ids = ranks_to_ids_incoming[source_rank];
181  const std::vector<int> &data = rank_to_incoming_data[source_rank];
182 
183  auto curr_id = incoming_ids.cbegin();
184  auto curr_global_index_iter = data.cbegin();
185  while (curr_id != incoming_ids.cend()) {
186  id_to_global_index[*curr_id] = *curr_global_index_iter;
187  curr_id++;
188  curr_global_index_iter++;
189  }
190  }
191 
192  // wait for all the sends to finsh
193  MPI_Waitall((int) send_requests.size(), send_requests.data(), MPI_STATUSES_IGNORE);
194 
195  // update global indexes in nbrinfo objects
196  for (auto &pinfo : pinfos) {
197  pinfo.setNeighborGlobalIndexes(id_to_global_index);
198  }
199  }
