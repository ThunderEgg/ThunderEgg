117  template <int D> static void GetRealCoord(const PatchInfo<D> &pinfo, const std::array<int, D> &coord, std::array<double, D> &real_coord)
118  {
119  loop<0, D - 1>([&](int dir) {
120  if (coord[dir] == -1) {
121  real_coord[dir] = pinfo.starts[dir];
122  } else if (coord[dir] == pinfo.ns[dir]) {
123  real_coord[dir] = pinfo.starts[dir] + pinfo.spacings[dir] * pinfo.ns[dir];
124  } else {
125  real_coord[dir] = pinfo.starts[dir] + pinfo.spacings[dir] / 2.0 + pinfo.spacings[dir] * coord[dir];
126  }
127  });
128  }

140  }
151  template <int D>
152  static void GetRealCoordBound(const PatchInfo<D> &pinfo, const std::array<int, D - 1> &coord, Side<D> s, std::array<double, D> &real_coord)
153  {
154  for (size_t dir = 0; dir < s.getAxisIndex(); dir++) {
155  if (coord[dir] == -1) {
156  real_coord[dir] = pinfo.starts[dir];
157  } else if (coord[dir] == pinfo.ns[dir]) {
158  real_coord[dir] = pinfo.starts[dir] + pinfo.spacings[dir] * pinfo.ns[dir];
159  } else {
160  real_coord[dir] = pinfo.starts[dir] + pinfo.spacings[dir] / 2.0 + pinfo.spacings[dir] * coord[dir];
161  }
162  }
163  if (s.isLowerOnAxis()) {
164  real_coord[s.getAxisIndex()] = pinfo.starts[s.getAxisIndex()];
165  } else {
166  real_coord[s.getAxisIndex()] = pinfo.starts[s.getAxisIndex()] + pinfo.spacings[s.getAxisIndex()] * pinfo.ns[s.getAxisIndex()];
167  }
168  for (size_t dir = s.getAxisIndex() + 1; dir < D; dir++) {
169  if (coord[dir - 1] == -1) {
170  real_coord[dir] = pinfo.starts[dir];
171  } else if (coord[dir - 1] == pinfo.ns[dir]) {
172  real_coord[dir] = pinfo.starts[dir] + pinfo.spacings[dir] * pinfo.ns[dir];
173  } else {
174  real_coord[dir] = pinfo.starts[dir] + pinfo.spacings[dir] / 2.0 + pinfo.spacings[dir] * coord[dir - 1];
175  }
176  }
177  }

188  static void
189  SetValues(const Domain<D> &domain, Vector<D> &vec, int component_index, std::function<double(const std::array<double, (int) D> &)> func)
190  {
191  if (component_index >= vec.getNumComponents()) {
192  throw RuntimeError("Invalid component to set");
193  }
194  std::array<double, D> real_coord;
195  for (int i = 0; i < vec.getNumLocalPatches(); i++) {
196  ComponentView<double, D> ld = vec.getComponentView(component_index, i);
197  auto pinfo = domain.getPatchInfoVector()[i];
198  nested_loop<D>(ld.getStart(), ld.getEnd(), [&](const std::array<int, D> &coord) {
199  GetRealCoord<D>(pinfo, coord, real_coord);
200  ld[coord] = func(real_coord);
201  });
202  }
203  }

449  template <int D> static double Integrate(const Domain<D> &domain, const Vector<D> &u)
450  {
451  double sum = 0;
452 
453  for (const auto &pinfo : domain.getPatchInfoVector()) {
454  for (int c = 0; c < u.getNumComponents(); c++) {
455  ComponentView<const double, D> u_data = u.getComponentView(c, pinfo.local_index);
456 
457  double patch_sum = 0;
458  nested_loop<D>(u_data.getStart(), u_data.getEnd(), [&](std::array<int, D> coord) { patch_sum += u_data[coord]; });
459 
460  for (size_t i = 0; i < D; i++) {
461  patch_sum *= pinfo.spacings[i];
462  }
463  sum += patch_sum;
464  }
465  }
466  double retval;
467  MPI_Allreduce(&sum, &retval, 1, MPI_DOUBLE, MPI_SUM, MPI_COMM_WORLD);
468  return retval;
469  }
