126  void setIncomingBufferMapsAndDetermineLocalVectorSize(const InterfaceDomain<D> &iface_domain)
127  {
128  int rank;
129  MPI_Comm_rank(MPI_COMM_WORLD, &rank);
130 
131  std::map<int, std::set<std::pair<int, int>>> incoming_ranks_to_id_local_index_pairs;
132 
133  for (auto piinfo : iface_domain.getPatchIfaceInfos()) {
134  for (Side<D> s : Side<D>::getValues()) {
135  if (piinfo->pinfo.hasNbr(s)) {
136  auto iface_info = piinfo->getIfaceInfo(s);
137  if (iface_info->rank != rank) {
138  incoming_ranks_to_id_local_index_pairs[iface_info->rank].emplace(iface_info->id, iface_info->patch_local_index);
139  }
140  }
141  }
142  }
143  int local_vector_size = 0;
144  for (auto rank_to_id_local_index_pairs : incoming_ranks_to_id_local_index_pairs) {
145  local_vector_size += rank_to_id_local_index_pairs.second.size();
146  }
147  local_vector_size += iface_domain.getNumLocalInterfaces();
148  num_local_patch_ifaces = local_vector_size;
149 
150  num_recvs = incoming_ranks_to_id_local_index_pairs.size();
151  recv_ranks.resize(num_recvs);
152  recv_local_indexes.resize(num_recvs);
153 
154  int recv_index = 0;
155  for (const auto &rank_to_id_local_index_pairs : incoming_ranks_to_id_local_index_pairs) {
156  recv_ranks[recv_index] = rank_to_id_local_index_pairs.first;
157  std::vector<int> &local_index_vector = recv_local_indexes[recv_index];
158 
159  local_index_vector.reserve(rank_to_id_local_index_pairs.second.size());
160 
161  for (auto id_local_index_pair : rank_to_id_local_index_pairs.second) {
162  local_index_vector.push_back(id_local_index_pair.second);
163  }
164  recv_index++;
165  }
166  }
172  void setOutgoingBufferMaps(const InterfaceDomain<D> &iface_domain)
173  {
174  int rank;
175  MPI_Comm_rank(MPI_COMM_WORLD, &rank);
176 
177  std::map<int, std::set<std::pair<int, int>>> outgoing_ranks_to_id_local_index_pairs;
178  for (auto iface : iface_domain.getInterfaces()) {
179  for (auto patch : iface->patches) {
180  if ((patch.type.isNormal() || patch.type.isFineToFine() || patch.type.isCoarseToCoarse()) && patch.piinfo->pinfo.rank != rank) {
181  outgoing_ranks_to_id_local_index_pairs[patch.piinfo->pinfo.rank].emplace(iface->id, iface->local_index);
182  }
183  }
184  }
185 
186  num_sends = outgoing_ranks_to_id_local_index_pairs.size();
187  send_ranks.resize(num_sends);
188  send_local_indexes.resize(num_sends);
189 
190  int send_index = 0;
191  for (const auto &rank_to_id_local_index_pairs : outgoing_ranks_to_id_local_index_pairs) {
192  send_ranks[send_index] = rank_to_id_local_index_pairs.first;
193  std::vector<int> &local_index_vector = send_local_indexes[send_index];
194 
195  local_index_vector.reserve(rank_to_id_local_index_pairs.second.size());
196 
197  for (auto id_local_index_pair : rank_to_id_local_index_pairs.second) {
198  local_index_vector.push_back(id_local_index_pair.second);
199  }
200  send_index++;
201  }
202  }

262  State scatterStart(const Vector<D - 1> &global_vector, Vector<D - 1> &local_patch_iface_vector) const
263  {
264  for (int i = 0; i < global_vector.getNumLocalPatches(); i++) {
265  auto global_data = global_vector.getComponentView(0, i);
266  auto local_data = local_patch_iface_vector.getComponentView(0, i);
267  nested_loop<D - 1>(
268  local_data.getStart(), local_data.getEnd(), [&](const std::array<int, D - 1> &coord) { local_data[coord] = global_data[coord]; });
269  }
270 
271  State state = initializeMPIBuffers();
272 
273  for (int recv_index = 0; recv_index < num_recvs; recv_index++) {
274  MPI_Irecv(state.ptr->recv_buffers[recv_index].data(),
275  state.ptr->recv_buffers[recv_index].size(),
276  MPI_DOUBLE,
277  recv_ranks[recv_index],
278  0,
279  MPI_COMM_WORLD,
280  &state.ptr->recv_requests[recv_index]);
281  }
282 
283  for (int send_index = 0; send_index < num_sends; send_index++) {
284  std::vector<double> &buffer = state.ptr->send_buffers[send_index];
285 
286  int buffer_index = 0;
287  for (int local_index : send_local_indexes[send_index]) {
288  auto local_data = global_vector.getComponentView(0, local_index);
289  nested_loop<D - 1>(local_data.getStart(), local_data.getEnd(), [&](const std::array<int, D - 1> &coord) {
290  buffer[buffer_index] = local_data[coord];
291  buffer_index++;
292  });
293  }
294 
295  MPI_Isend(buffer.data(), buffer.size(), MPI_DOUBLE, send_ranks[send_index], 0, MPI_COMM_WORLD, &state.ptr->send_requests[send_index]);
296  }
297 
298  for (int local_iface = 0; local_iface < global_vector.getNumLocalPatches(); local_iface++) {
299  auto global_data = global_vector.getComponentView(0, local_iface);
300  auto local_data = local_patch_iface_vector.getComponentView(0, local_iface);
301  nested_loop<D - 1>(
302  local_data.getStart(), local_data.getEnd(), [&](const std::array<int, D - 1> &coord) { local_data[coord] = global_data[coord]; });
303  }
304 
305  state.ptr->curr_global_vector = &global_vector;
306  state.ptr->curr_local_vector = &local_patch_iface_vector;
307  return state;
308  }
319  void scatterFinish(const State &state, const Vector<D - 1> &global_vector, Vector<D - 1> &local_patch_iface_vector) const
320  {
321  if (&global_vector != state.ptr->curr_global_vector || &local_patch_iface_vector != state.ptr->curr_local_vector) {
322  throw RuntimeError("Different vectors were passed ot scatterFinish than were passed to scatterStart");
323  }
324 
325  for (int i = 0; i < num_recvs; i++) {
326  MPI_Status status;
327  int recv_index;
328  MPI_Waitany(num_recvs, state.ptr->recv_requests.data(), &recv_index, &status);
329 
330  std::vector<double> &buffer = state.ptr->recv_buffers[recv_index];
331 
332  int buffer_index = 0;
333  for (int local_index : recv_local_indexes[recv_index]) {
334  auto local_data = local_patch_iface_vector.getComponentView(0, local_index);
335  nested_loop<D - 1>(local_data.getStart(), local_data.getEnd(), [&](const std::array<int, D - 1> &coord) {
336  local_data[coord] = buffer[buffer_index];
337  buffer_index++;
338  });
339  }
340  }
341 
342  MPI_Waitall(num_sends, state.ptr->send_requests.data(), MPI_STATUSES_IGNORE);
343 
344  state.ptr->communicating = false;
345  }
