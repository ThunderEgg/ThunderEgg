375  static std::array<Orthant, num_orthants / 2> getValuesOnSide(Side<D> s)
376  {
377  unsigned int bit_to_insert = s.getAxisIndex();
378  unsigned int set_bit = s.isLowerOnAxis() ? 0 : 1;
379  unsigned int lower_mask = ~((~0x0U) << bit_to_insert);
380  unsigned int upper_mask = (~0x0U) << (bit_to_insert + 1);
381 
382  std::array<Orthant<D>, Orthant<D>::num_orthants / 2> retval;
383  for (size_t i = 0; i < Orthant<D>::num_orthants / 2; i++) {
384  size_t value = (i << 1) & upper_mask;
385  value |= i & lower_mask;
386  value |= set_bit << bit_to_insert;
387  retval[i] = Orthant<D>(value);
388  }
389  return retval;
390  }
