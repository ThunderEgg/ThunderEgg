94  template <int M> SliceInfo<M> getSliceOnPriv(Face<D, M> f, const std::array<int, D - M> &offset) const
95  {
96  SliceInfo<M> info;
97 
98  info.first_value = this->getGhostStart();
99 
100  std::array<Side<D>, D - M> sides = f.getSides();
101  size_t lengths_index = 0;
102  size_t sides_index = 0;
103 
104  for (size_t axis = 0; axis < (size_t) D; axis++) {
105  if (sides_index < sides.size() && sides[sides_index].getAxisIndex() == axis) {
106  if (sides[sides_index].isLowerOnAxis()) {
107  info.first_value[axis] = offset[sides_index];
108  } else {
109  info.first_value[axis] = this->getEnd()[axis] - offset[sides_index];
110  }
111  sides_index++;
112  } else {
113  info.strides[lengths_index] = this->getStrides()[axis];
114  info.ghost_start[lengths_index] = this->getGhostStart()[axis];
115  info.start[lengths_index] = this->getStart()[axis];
116  info.end[lengths_index] = this->getEnd()[axis];
117  info.ghost_end[lengths_index] = this->getGhostEnd()[axis];
118  lengths_index++;
119  }
120  }
121  info.strides[M] = this->getStrides()[D];
122  info.ghost_start[M] = this->getGhostStart()[D];
123  info.start[M] = this->getStart()[D];
124  info.end[M] = this->getEnd()[D];
125  info.ghost_end[M] = this->getGhostEnd()[D];
126  return info;
127  }

200  template <int M> View<typename std::remove_const<T>::type, M + 1> getGhostSliceOn(Face<D, M> f, const std::array<size_t, D - M> &offset) const
201  {
202  using noconst_T = typename std::remove_const<T>::type;
203  using noconst_T_ptr = typename std::add_pointer<noconst_T>::type;
204  std::array<int, M + 1> new_strides;
205  std::array<int, M + 1> new_ghost_start;
206  std::array<int, M + 1> new_start;
207  std::array<int, M + 1> new_end;
208  std::array<int, M + 1> new_ghost_end;
209  std::array<int, D + 1> first_value;
210 
211  first_value = this->getGhostStart();
212 
213  std::array<Side<D>, D - M> sides = f.getSides();
214  size_t lengths_index = 0;
215  size_t sides_index = 0;
216 
217  for (size_t axis = 0; axis < (size_t) D; axis++) {
218  if (sides_index < sides.size() && sides[sides_index].getAxisIndex() == axis) {
219  if (sides[sides_index].isLowerOnAxis()) {
220  first_value[axis] = -1 - offset[sides_index];
221  } else {
222  first_value[axis] = this->getEnd()[axis] + 1 + offset[sides_index];
223  }
224  sides_index++;
225  } else {
226  new_strides[lengths_index] = this->getStrides()[axis];
227  new_ghost_start[lengths_index] = this->getGhostStart()[axis];
228  new_start[lengths_index] = this->getStart()[axis];
229  new_end[lengths_index] = this->getEnd()[axis];
230  new_ghost_end[lengths_index] = this->getGhostEnd()[axis];
231  lengths_index++;
232  }
233  }
234  new_strides[M] = this->getStrides()[D];
235  new_ghost_start[M] = this->getGhostStart()[D];
236  new_start[M] = this->getStart()[D];
237  new_end[M] = this->getEnd()[D];
238  new_ghost_end[M] = this->getGhostEnd()[D];
239 
240  noconst_T_ptr new_data = const_cast<noconst_T_ptr>(&(*this)[first_value]); // Thunderegg doesn't care if values in ghosts are modified
241  return View<noconst_T, M + 1>(new_data, new_strides, new_ghost_start, new_start, new_end, new_ghost_end);
242  }

244  ComponentView<T, D> getComponentView(int component_index) const
245  {
246  if constexpr (ENABLE_DEBUG) {
247  if (component_index < this->getStart()[D] || component_index > this->getEnd()[D]) {
248  throw RuntimeError("invalid component index");
249  }
250  }
251  std::array<int, D> new_strides;
252  std::array<int, D> new_ghost_start;
253  std::array<int, D> new_start;
254  std::array<int, D> new_end;
255  std::array<int, D> new_ghost_end;
256  std::array<int, D + 1> first_value;
257 
258  first_value = this->getGhostStart();
259  first_value[D] = component_index;
260 
261  for (size_t axis = 0; axis < (size_t) D; axis++) {
262  new_strides[axis] = this->getStrides()[axis];
263  new_ghost_start[axis] = this->getGhostStart()[axis];
264  new_start[axis] = this->getStart()[axis];
265  new_end[axis] = this->getEnd()[axis];
266  new_ghost_end[axis] = this->getGhostEnd()[axis];
267  }
268 
269  T_ptr new_data = (&(*this)[first_value]);
270  return ComponentView<T, D>(new_data, new_strides, new_ghost_start, new_start, new_end, new_ghost_end);
