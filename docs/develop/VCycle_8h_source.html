33 template <int D> class VCycle : public Cycle<D>
34 {
35  private:
36  int num_pre_sweeps = 1;
37  int num_post_sweeps = 1;
38  int num_coarse_sweeps = 1;
39 
40  protected:
41  /*
42  * @brief Implements V-cycle. Pre-smooth, visit coarser level and then post-smooth.
43  */
44  void visit(const Level<D> &level, const Vector<D> &f, Vector<D> &u) const override
45  {
46  if (level.coarsest()) {
47  for (int i = 0; i < num_coarse_sweeps; i++) {
48  level.getSmoother().smooth(f, u);
49  }
50  } else {
51  for (int i = 0; i < num_pre_sweeps; i++) {
52  level.getSmoother().smooth(f, u);
53  }
54 
55  Vector<D> coarser_f = this->restrict(level, f, u);
56 
57  const Level<D> &coarser_level = *level.getCoarser();
58  Vector<D> coarser_u = coarser_f.getZeroClone();
59 
60  this->visit(coarser_level, coarser_f, coarser_u);
61 
62  coarser_level.getInterpolator().interpolate(coarser_u, u);
63 
64  for (int i = 0; i < num_post_sweeps; i++) {
65  level.getSmoother().smooth(f, u);
66  }
67  }
68  }
69 
70  public:
76  VCycle(std::shared_ptr<Level<D>> finest_level, const CycleOpts &opts) : Cycle<D>(finest_level)
77  {
78  num_pre_sweeps = opts.pre_sweeps;
79  num_post_sweeps = opts.post_sweeps;
80  num_coarse_sweeps = opts.coarse_sweeps;
81  }
87  VCycle<D> *clone() const override
88  {
89  return new VCycle(*this);
90  }
91 };
