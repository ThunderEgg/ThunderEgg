93  std::array<fftw_r2r_kind, D> getTransformsForPatch(const PatchInfo<D>& pinfo)
94  {
95  // get transform types for each axis
96  std::array<fftw_r2r_kind, D> transforms;
97  for (size_t axis = 0; axis < D; axis++) {
98  if (patchIsNeumannOnSide(pinfo, LowerSideOnAxis<D>(axis)) &&
99  patchIsNeumannOnSide(pinfo, HigherSideOnAxis<D>(axis))) {
100  transforms[D - 1 - axis] = FFTW_REDFT10;
101  } else if (patchIsNeumannOnSide(pinfo, LowerSideOnAxis<D>(axis))) {
102  transforms[D - 1 - axis] = FFTW_REDFT11;
103  } else if (patchIsNeumannOnSide(pinfo, HigherSideOnAxis<D>(axis))) {
104  transforms[D - 1 - axis] = FFTW_RODFT11;
105  } else {
106  transforms[D - 1 - axis] = FFTW_RODFT10;
107  }

142  PatchArray<D> getEigenValues(const PatchInfo<D>& pinfo)
143  {
144  PatchArray<D> retval(this->getDomain().getNs(), 1, 0);
145 
146  std::valarray<size_t> all_strides(D);
147  size_t curr_stride = 1;
148  for (size_t i = 0; i < D; i++) {
149  all_strides[i] = curr_stride;
150  curr_stride *= pinfo.ns[i];
151  }
152 
153  for (size_t axis = 0; axis < D; axis++) {
154  int n = pinfo.ns[axis];
155  double h = pinfo.spacings[axis];
156 
157  if (patchIsNeumannOnSide(pinfo, LowerSideOnAxis<D>(axis)) &&
158  patchIsNeumannOnSide(pinfo, HigherSideOnAxis<D>(axis))) {
159  for (int xi = 0; xi < n; xi++) {
160  double val = 4 / (h * h) * pow(sin(xi * M_PI / (2 * n)), 2);
161  View<double, D> slice = retval.getSliceOn(Side<D>(2 * axis), { xi });
162  Loop::OverInteriorIndexes<D>(
163  slice, [&](const std::array<int, D>& coord) { slice[coord] -= val; });
164  }
165  } else if (patchIsNeumannOnSide(pinfo, LowerSideOnAxis<D>(axis)) ||
166  patchIsNeumannOnSide(pinfo, HigherSideOnAxis<D>(axis))) {
167  for (int xi = 0; xi < n; xi++) {
168  double val = 4 / (h * h) * pow(sin((xi + 0.5) * M_PI / (2 * n)), 2);
169  View<double, D> slice = retval.getSliceOn(Side<D>(2 * axis), { xi });
170  Loop::OverInteriorIndexes<D>(
171  slice, [&](const std::array<int, D>& coord) { slice[coord] -= val; });
172  }
173  } else {
174  for (int xi = 0; xi < n; xi++) {
175  double val = 4 / (h * h) * pow(sin((xi + 1) * M_PI / (2 * n)), 2);
176  View<double, D> slice = retval.getSliceOn(Side<D>(2 * axis), { xi });
177  Loop::OverInteriorIndexes<D>(
178  slice, [&](const std::array<int, D>& coord) { slice[coord] -= val; });
179  }
180  }
181  }
182  return retval;
183  }

192  FFTWPatchSolver(const PatchOperator<D>& op, std::bitset<Side<D>::number_of> neumann)

197  CompareFunction compare = [&](const PatchInfo<D>& a, const PatchInfo<D>& b) {
198  std::bitset<Side<D>::number_of> a_neumann;
199  std::bitset<Side<D>::number_of> b_neumann;
200  for (Side<D> s : Side<D>::getValues()) {
201  a_neumann[s.getIndex()] = patchIsNeumannOnSide(a, s);
202  b_neumann[s.getIndex()] = patchIsNeumannOnSide(b, s);
203  }
204  return std::forward_as_tuple(a_neumann.to_ulong(), a.spacings[0]) <
205  std::forward_as_tuple(b_neumann.to_ulong(), b.spacings[0]);
206  };

223  void solveSinglePatch(const PatchInfo<D>& pinfo,
224  const PatchView<const double, D>& f_view,
225  const PatchView<double, D>& u_view) const override
226  {
227  PatchArray<D> f_copy(pinfo.ns, 1, 0);
228  PatchArray<D> tmp(pinfo.ns, 1, 0);
229  PatchArray<D> sol(pinfo.ns, 1, 0);
230 
231  Loop::OverInteriorIndexes<D + 1>(
232  f_copy, [&](std::array<int, D + 1> coord) { f_copy[coord] = f_view[coord]; });
233 
234  op->modifyRHSForInternalBoundaryConditions(pinfo, u_view, f_copy.getView());
235 
236  fftw_execute_r2r(*plan1.at(pinfo), &f_copy[f_copy.getStart()], &tmp[tmp.getStart()]);
237 
238  const PatchArray<D>& eigen_vals_view = eigen_vals.at(pinfo);
239  Loop::OverInteriorIndexes<D + 1>(
240  tmp, [&](std::array<int, D + 1> coord) { tmp[coord] /= eigen_vals_view[coord]; });
241 
242  if (neumann.all() && !pinfo.hasNbr()) {
243  tmp[tmp.getStart()] = 0;
244  }
245 
246  fftw_execute_r2r(*plan2.at(pinfo), &tmp[tmp.getStart()], &sol[sol.getStart()]);
247 
248  double scale = 1;
249  for (size_t axis = 0; axis < D; axis++) {
250  scale *= 2.0 * this->getDomain().getNs()[axis];
251  }
252  Loop::OverInteriorIndexes<D + 1>(
253  u_view, [&](std::array<int, D + 1> coord) { u_view[coord] = sol[coord] / scale; });
254  }
262  void addPatch(const PatchInfo<D>& pinfo)
263  {
264  if (plan1.count(pinfo) == 0) {
265  // revers ns because FFTW is row major
266  std::array<int, D> ns_reversed;
267  for (size_t i = 0; i < D; i++) {
268  ns_reversed[D - 1 - i] = pinfo.ns[i];
269  }
270  std::array<fftw_r2r_kind, D> transforms = getTransformsForPatch(pinfo);
271  std::array<fftw_r2r_kind, D> transforms_inv = getInverseTransformsForPatch(pinfo);
272 
273  PatchArray<D> f_copy(pinfo.ns, 1, 0);
274  PatchArray<D> tmp(pinfo.ns, 1, 0);
275  PatchArray<D> sol(pinfo.ns, 1, 0);
276 
277  fftw_plan* fftw_plan1 = new fftw_plan();
278 
279  *fftw_plan1 = fftw_plan_r2r(D,
280  ns_reversed.data(),
281  &f_copy[f_copy.getStart()],
282  &tmp[tmp.getStart()],
283  transforms.data(),
284  FFTW_MEASURE | FFTW_DESTROY_INPUT | FFTW_UNALIGNED);
285 
286  plan1[pinfo] = std::shared_ptr<fftw_plan>(fftw_plan1, [](fftw_plan* plan) {
287  fftw_destroy_plan(*plan);
288  delete plan;
289  });
290 
291  fftw_plan* fftw_plan2 = new fftw_plan();
292 
293  *fftw_plan2 = fftw_plan_r2r(D,
294  ns_reversed.data(),
295  &tmp[tmp.getStart()],
296  &sol[sol.getStart()],
297  transforms_inv.data(),
298  FFTW_MEASURE | FFTW_DESTROY_INPUT | FFTW_UNALIGNED);
299 
300  plan2[pinfo] = std::shared_ptr<fftw_plan>(fftw_plan2, [](fftw_plan* plan) {
301  fftw_destroy_plan(*plan);
302  delete plan;
303  });
304 
305  eigen_vals.emplace(pinfo, getEigenValues(pinfo));
306  }
307  }
