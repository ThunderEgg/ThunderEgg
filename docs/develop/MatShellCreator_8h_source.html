66  static int applyMat(Mat A, Vec x, Vec b)
67  {
68  MatShellCreator<D> *msc = nullptr;
69  MatShellGetContext(A, &msc);
70 
71  Vector<D> te_x = msc->getNewVector();
72  Vector<D> te_b = msc->getNewVector();
73 
74  // petsc vectors don't have gost padding for patchs, so this is neccesary
75  const double *x_view;
76  VecGetArrayRead(x, &x_view);
77  int index = 0;
78  for (int p_index = 0; p_index < te_x.getNumLocalPatches(); p_index++) {
79  for (int c = 0; c < te_x.getNumComponents(); c++) {
80  ComponentView<double, D> ld = te_x.getComponentView(c, p_index);
81  nested_loop<D>(ld.getStart(), ld.getEnd(), [&](const std::array<int, D> &coord) {
82  ld[coord] = x_view[index];
83  index++;
84  });
85  }
86  }
87 
88  VecRestoreArrayRead(x, &x_view);
89 
90  msc->op->apply(te_x, te_b);
91 
92  double *b_view;
93  VecGetArray(b, &b_view);
94  index = 0;
95  for (int p_index = 0; p_index < te_b.getNumLocalPatches(); p_index++) {
96  for (int c = 0; c < te_b.getNumComponents(); c++) {
97  const ComponentView<const double, D> ld = te_b.getComponentView(c, p_index);
98  nested_loop<D>(ld.getStart(), ld.getEnd(), [&](const std::array<int, D> &coord) {
99  b_view[index] = ld[coord];
100  index++;
101  });
102  }
103  }
104  VecRestoreArray(b, &b_view);
105 
106  return 0;
107  }
114  static int destroyMat(Mat A)
115  {
116  MatShellCreator<D> *msc = nullptr;
117  MatShellGetContext(A, &msc);
118  delete msc;
119  return 0;
120  }

130  static Mat GetNewMatShell(const Operator<D> &op, const std::function<Vector<D>()> &vector_allocator)
131  {
132  MatShellCreator<D> *msc = new MatShellCreator(op, vector_allocator);
133  Vector<D> vec = vector_allocator();
134  int m = vec.getNumLocalCells() * vec.getNumComponents();
135  Mat A;
136  MatCreateShell(MPI_COMM_WORLD, m, m, PETSC_DETERMINE, PETSC_DETERMINE, msc, &A);
137  MatShellSetOperation(A, MATOP_MULT, (void (*)(void)) applyMat);
138  MatShellSetOperation(A, MATOP_DESTROY, (void (*)(void)) destroyMat);
139  return A;
140  }
