146  new_level->setOperator(op);
147  new_level->setSmoother(smoother);

149

151  prev_level->setCoarser(new_level);
152  }
159  std::shared_ptr<Cycle<D>> getCycle() const
160  {
161  if (!has_coarsest) {
162  throw RuntimeError("addCoarsestLevel has not been called");
163  }
164  std::shared_ptr<Cycle<D>> cycle;
165  if (opts.cycle_type == "V") {
166  cycle.reset(new VCycle<D>(finest_level, opts));
167  } else if (opts.cycle_type == "W") {
168  cycle.reset(new WCycle<D>(finest_level, opts));
169  } else if (opts.cycle_type == "F") {
170  cycle.reset(new FMGCycle<D>(finest_level, opts));
171  } else {
172  throw RuntimeError("Unsupported Cycle type: " + opts.cycle_type);
173  }
174  return cycle;
175  }
