497  std::array<Face<D, D - 1>, D - M> getSides() const
498  {
499  std::array<Face<D, D - 1>, D - M> sides;
500  if constexpr (D > 1 && M == 0) {
501  for (size_t i = 0; i < D; i++) {
502  size_t side = 2 * i;
503  if ((1 << i) & value) {
504  side |= 1;
505  }
506  sides[i] = Face<D, D - 1>(side);
507  }
508  } else if constexpr (D == 3 && M == 1) {
509  size_t not_axis = value >> 2;
510  size_t curr_index = 0;
511  for (unsigned char i = 0; i < D; i++) {
512  if (i != not_axis) {
513  unsigned char bit = (value & (0b1 << curr_index)) >> curr_index;
514  sides[curr_index] = Face<D, D - 1>((i << 1) ^ bit);
515  curr_index++;
516  }
517  }
518  } else {
519  sides[0] = Face<D, D - 1>(value);
520  }
521  return sides;
522  }
