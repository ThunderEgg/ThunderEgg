327  template <int M> class GhostViewInfo
328  {
329  private:
333  std::array<std::array<int, D + 1>, Face<D, M>::number_of> strides;
337  std::array<std::array<int, D + 1>, Face<D, M>::number_of> ghost_start;
341  std::array<std::array<int, D + 1>, Face<D, M>::number_of> start;
345  std::array<std::array<int, D + 1>, Face<D, M>::number_of> end;
349  std::array<std::array<int, D + 1>, Face<D, M>::number_of> ghost_end;
353  std::array<size_t, Face<D, M>::number_of> sizes;
354 
355  public:
359  GhostViewInfo() = default;
365  explicit GhostViewInfo(const Domain<D> &domain)
366  {
367  std::array<int, D + 1> ns;
368  for (int i = 0; i < D; i++) {
369  ns[i] = domain.getNs()[i];
370  }
371  ns[D] = 1;
372  int num_ghost_cells = domain.getNumGhostCells();
373  for (Face<D, M> face : Face<D, M>::getValues()) {
374  std::array<int, D + 1> ghost_ns = ns;
375 
376  std::array<int, D + 1> face_ghost_start;
377  face_ghost_start.fill(0);
378 
379  std::array<int, D + 1> face_start;
380  face_start.fill(0);
381 
382  std::array<int, D + 1> face_end;
383  face_end = ns;
384  for (int &v : face_end) {
385  v--;
386  }
387 
388  std::array<int, D + 1> face_ghost_end;
389  face_ghost_end = ns;
390  for (int &v : face_ghost_end) {
391  v--;
392  }
393 
394  for (Side<D> side : face.getSides()) {
395  ghost_ns[side.getAxisIndex()] = num_ghost_cells;
396  if (side.isLowerOnAxis()) {
397  face_ghost_start[side.getAxisIndex()] = -num_ghost_cells;
398  face_ghost_end[side.getAxisIndex()] = -1;
399  face_end[side.getAxisIndex()] = -1;
400  } else {
401  face_ghost_start[side.getAxisIndex()] = ns[side.getAxisIndex()];
402  face_start[side.getAxisIndex()] = ns[side.getAxisIndex()];
403  face_ghost_end[side.getAxisIndex()] = ns[side.getAxisIndex()] + num_ghost_cells - 1;
404  }
405  }
406  strides[face.getIndex()][0] = 1;
407  for (size_t i = 1; i < D + 1; i++) {
408  strides[face.getIndex()][i] = ghost_ns[i - 1] * strides[face.getIndex()][i - 1];
409  }
410  sizes[face.getIndex()] = ghost_ns[D - 1] * strides[face.getIndex()][D - 1];
411  int offset = 0;
412  for (Side<D> side : face.getSides()) {
413  if (side.isHigherOnAxis()) {
414  offset += -(ns[side.getAxisIndex()] + num_ghost_cells) * strides[face.getIndex()][side.getAxisIndex()];
415  }
416  }
417  ghost_start[face.getIndex()] = face_ghost_start;
418  start[face.getIndex()] = face_start;
419  end[face.getIndex()] = face_end;
420  ghost_end[face.getIndex()] = face_ghost_end;
421  }
422  }

496  template <int M> void addRecvBufferToGhost(const RemoteCallSet &remote_call_set, std::vector<double> &buffer, const Vector<D> &u) const
497  {
498  for (const IncomingGhost<M> &incoming_ghost : remote_call_set.incoming_ghosts.template get<M>()) {
499  int local_index = incoming_ghost.local_index;
500  Face<D, M> face = incoming_ghost.face;
501  size_t buffer_offset = incoming_ghost.offset;
502 
503  PatchView<const double, D> local_view = u.getPatchView(local_index);
504  double * buffer_ptr = buffer.data() + buffer_offset * u.getNumComponents();
505  PatchView<const double, D> buffer_view = getPatchViewForBuffer(buffer_ptr, face, u.getNumComponents());
506  std::array<size_t, D - M> start;
507  start.fill(0);
508  std::array<size_t, D - M> end;
509  end.fill(domain.getNumGhostCells() - 1);
510  nested_loop<D - M>(start, end, [&](const std::array<size_t, D - M> &offset) {
511  View<double, M + 1> local_slice = local_view.getGhostSliceOn(face, offset);
512  View<double, M + 1> buffer_slice = buffer_view.getGhostSliceOn(face, offset);
513  loop_over_interior_indexes<M + 1>(local_slice,
514  [&](const std::array<int, M + 1> &coord) { local_slice[coord] += buffer_slice[coord]; });
515  });
516  }
517  }
525  void processRecvs(std::vector<MPI_Request> &requests, std::vector<std::vector<double>> &buffers, const Vector<D> &u) const
526  {
527  size_t num_requests = requests.size();
528  for (size_t i = 0; i < num_requests; i++) {
529  int finished_index;
530  MPI_Waitany(requests.size(), requests.data(), &finished_index, MPI_STATUS_IGNORE);
531  switch (fill_type) {
532  case GhostFillingType::Corners:
533  if constexpr (D >= 2) {
534  addRecvBufferToGhost<0>(remote_call_sets[finished_index], buffers[finished_index], u);
535  }
536  case GhostFillingType::Edges:
537  if constexpr (D == 3) {
538  addRecvBufferToGhost<1>(remote_call_sets[finished_index], buffers[finished_index], u);
539  }
540  case GhostFillingType::Faces:
541  addRecvBufferToGhost<D - 1>(remote_call_sets[finished_index], buffers[finished_index], u);
542  break;
543  default:
544  throw RuntimeError("Unsupported GhostFilling Type");
545  }
546  }

746  template <int M> void enumerateCalls(std::deque<LocalCall<M>> &my_local_calls, std::map<int, RemoteCallSet> &rank_to_remote_call_sets) const
747  {
748  int rank;
749  MPI_Comm_rank(MPI_COMM_WORLD, &rank);
750  std::map<int, std::set<RemoteCallPrototype<M>>> rank_to_remote_call_prototypes;
751  std::map<int, std::set<IncomingGhostPrototype<M>>> rank_to_incoming_ghost_prototypes;
752 
753  for (const PatchInfo<D> &pinfo : domain.getPatchInfoVector()) {
754  for (Face<D, M> f : Face<D, M>::getValues()) {
755  if (pinfo.hasNbr(f)) {
756  switch (pinfo.getNbrType(f)) {
757  case NbrType::Normal:
758  addNormalNbrCalls(pinfo, f, my_local_calls, rank_to_remote_call_prototypes, rank_to_incoming_ghost_prototypes);
759  break;
760  case NbrType::Fine:
761  addFineNbrCalls(pinfo, f, my_local_calls, rank_to_remote_call_prototypes, rank_to_incoming_ghost_prototypes);
762  break;
763  case NbrType::Coarse:
764  addCoarseNbrCalls(pinfo, f, my_local_calls, rank_to_remote_call_prototypes, rank_to_incoming_ghost_prototypes);
765  break;
766  default:
767  throw RuntimeError("Unsupported NbrType");
768  }
769  }
770  }
771  }
772  const GhostViewInfo<M> &ghost_local_data_info = ghost_local_data_infos.template get<M>();
773  for (const auto &pair : rank_to_remote_call_prototypes) {
774  int rank = pair.first;
775  rank_to_remote_call_sets.emplace(rank, rank);
776  RemoteCallSet & remote_call_set = rank_to_remote_call_sets.at(rank);
777  std::tuple<int, Face<D, M>> prev_id_side;
778  for (const RemoteCallPrototype<M> &call : pair.second) {
779  size_t offset = remote_call_set.send_buffer_length;
780 
781  // calculate length in buffer need for ghost cells
782  size_t length = ghost_local_data_info.getSize(call.face);
783  // add length to buffer length
784  // if its the same side of the patch, resuse the previous buffer space
785  if (std::make_tuple(call.id, call.face) == prev_id_side) {
786  offset -= length;
787  } else {
788  remote_call_set.send_buffer_length += length;
789  }
790  remote_call_set.remote_calls.template get<M>().emplace_back(call, offset);
791  prev_id_side = std::make_tuple(call.id, call.face);
792  }
793  }
794  for (const auto &pair : rank_to_incoming_ghost_prototypes) {
795  RemoteCallSet &remote_call_set = rank_to_remote_call_sets.at(pair.first);
796  for (const IncomingGhostPrototype<M> &prototype : pair.second) {
797  // add length for ghosts to buffer length
798  size_t length = ghost_local_data_info.getSize(prototype.face);
799  size_t offset = remote_call_set.recv_buffer_length;
800  remote_call_set.recv_buffer_length += length;
801 
802  // add ghost to incoming ghosts
803  remote_call_set.incoming_ghosts.template get<M>().emplace_back(prototype, offset);
804  }
805  }
806  }
