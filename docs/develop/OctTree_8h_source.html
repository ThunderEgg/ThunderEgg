95 template <int D> inline Tree<D>::Tree(std::string file_name)
96 {
97  using namespace std;
98  ifstream input(file_name, ios_base::binary);
99  int num_nodes;
100  int num_trees;
101  input.read((char *) &num_nodes, 4);
102  input.read((char *) &num_trees, 4);
103  num_levels = 0;
104  max_id = 0;
105  for (int i = 0; i < num_nodes; i++) {
106  Node<D> n;
107  // id level parent
108  input.read((char *) &n.id, 3 * 4);
109  // length and starts
110  input.read((char *) &n.lengths[0], sizeof(n.lengths));
111  input.read((char *) &n.starts[0], sizeof(n.starts));
112 
113  input.read((char *) &n.nbr_id[0], sizeof(n.nbr_id));
114  input.read((char *) &n.child_id[0], sizeof(n.child_id));
115 
116  if (i == 0) {
117  root = n.id;
118  }
119  max_id = max(max_id, n.id);
120  nodes[n.id] = n;
121 
122  if (n.level > num_levels) {
123  num_levels = n.level;
124  }
125  levels[n.level] = &nodes[n.id];
126  }

128 template <int D> inline void Tree<D>::refineLeaves()
129 {
130  using namespace std;
131  Node<D> root_node = nodes[root];
132  Node<D> child = root_node;
133  int level = 0;
134  while (child.hasChildren()) {
135  child = nodes[child.child_id[0]];
136  level++;
137  }
138  deque<pair<int, int>> q;
139  set<pair<int, int>> qed;
140  q.push_back(make_pair(level, child.id));
141  qed.insert(make_pair(level, child.id));
142 
143  while (!q.empty()) {
144  pair<int, int> p = q.front();
145  Node<D> n = nodes[p.second];
146  int level = p.first;
147  q.pop_front();
148 
149  // set and enqueue nbrs
150  for (Side<D> s : Side<D>::getValues()) {
151  // coarser
152  if (n.nbrId(s) == -1 && n.parent != -1 && nodes[n.parent].nbrId(s) != -1) {
153  Node<D> parent = nodes[n.parent];
154  Node<D> nbr = nodes[parent.nbrId(s)];
155  pair<int, int> p(level - 1, nbr.id);
156  if (!qed.count(p)) {
157  q.push_back(p);
158  qed.insert(p);
159  }
160  // finer
161  } else if (n.nbrId(s) != -1 && nodes[n.nbrId(s)].hasChildren()) {
162  Node<D> nbr = nodes[n.nbrId(s)];
163  auto octs = Orthant<D>::getValuesOnSide(s.opposite());
164  for (size_t i = 0; i < octs.size(); i++) {
165  int id = nbr.childId(octs[i]);
166  pair<int, int> p(level + 1, id);
167  if (!qed.count(p)) {
168  q.push_back(p);
169  qed.insert(p);
170  }
171  }
172  // normal
173  } else if (n.nbrId(s) != -1) {
174  int id = n.nbrId(s);
175  pair<int, int> p(level, id);
176  if (!qed.count(p)) {
177  q.push_back(p);
178  qed.insert(p);
179  }
180  }
181  }
182  }
183  for (auto p : qed) {
184  refineNode(nodes[p.second]);
185  }
186  levels[num_levels + 1] = &nodes.at(levels[num_levels]->child_id[0]);
187  this->num_levels++;
188 }
189 template <int D> inline void Tree<D>::refineNode(Node<D> &n)
190 {
191  std::array<Node<D>, Orthant<D>::num_orthants> new_children;
192  for (Orthant<D> o : Orthant<D>::getValues()) {
193  new_children[o.getIndex()] = Node<D>(n, o);
194  Node<D> &child = new_children[o.getIndex()];
195  max_id++;
196  child.id = max_id;
197  n.child_id[o.getIndex()] = max_id;
198  }
199  // set new neighbors
200  for (Orthant<D> o : Orthant<D>::getValues()) {
201  for (Side<D> s : o.getInteriorSides()) {
202  new_children[o.getIndex()].nbrId(s) = new_children[o.getNbrOnSide(s).getIndex()].id;
203  }
204  }
205 
206  // set outer neighbors
207  for (Side<D> s : Side<D>::getValues()) {
208  if (n.hasNbr(s) && nodes[n.nbrId(s)].hasChildren()) {
209  Node<D> &nbr = nodes[n.nbrId(s)];
210  for (Orthant<D> o : Orthant<D>::getValuesOnSide(s)) {
211  Node<D> &child = new_children[o.getIndex()];
212  Node<D> &nbr_child = nodes[nbr.childId(o.getNbrOnSide(s))];
213  child.nbrId(s) = nbr_child.id;
214  nbr_child.nbrId(s.opposite()) = child.id;
215  }
216  }
217  }
218  // add nodes
219  for (Node<D> child : new_children) {
220  nodes[child.id] = child;
221  }
222 }
