94  template <int M> SliceInfo<M> getSliceOnPriv(Face<D, M> f, const std::array<int, D - M> &offset) const
95  {
96  SliceInfo<M> info;
97 
98  info.first_value = this->getGhostStart();
99 
100  std::array<Side<D>, D - M> sides = f.getSides();
101  size_t lengths_index = 0;
102  size_t sides_index = 0;
103 
104  for (size_t axis = 0; axis < (size_t) D; axis++) {
105  if (sides_index < sides.size() && sides[sides_index].getAxisIndex() == axis) {
106  if (sides[sides_index].isLowerOnAxis()) {
107  info.first_value[axis] = offset[sides_index];
108  } else {
109  info.first_value[axis] = this->getEnd()[axis] - offset[sides_index];
110  }
111  sides_index++;
112  } else {
113  info.strides[lengths_index] = this->getStrides()[axis];
114  info.ghost_start[lengths_index] = this->getGhostStart()[axis];
115  info.start[lengths_index] = this->getStart()[axis];
116  info.end[lengths_index] = this->getEnd()[axis];
117  info.ghost_end[lengths_index] = this->getGhostEnd()[axis];
118  lengths_index++;
119  }
120  }
121  return info;
122  }

195  template <int M> View<typename std::remove_const<T>::type, M> getGhostSliceOn(Face<D, M> f, const std::array<size_t, D - M> &offset) const
196  {
197  using noconst_T = typename std::remove_const<T>::type;
198  using noconst_T_ptr = typename std::add_pointer<noconst_T>::type;
199  std::array<int, M> new_strides;
200  std::array<int, M> new_ghost_start;
201  std::array<int, M> new_start;
202  std::array<int, M> new_end;
203  std::array<int, M> new_ghost_end;
204  std::array<int, D> first_value;
205 
206  first_value = this->getGhostStart();
207 
208  std::array<Side<D>, D - M> sides = f.getSides();
209  size_t lengths_index = 0;
210  size_t sides_index = 0;
211 
212  for (size_t axis = 0; axis < (size_t) D; axis++) {
213  if (sides_index < sides.size() && sides[sides_index].getAxisIndex() == axis) {
214  if (sides[sides_index].isLowerOnAxis()) {
215  first_value[axis] = -1 - offset[sides_index];
216  } else {
217  first_value[axis] = this->getEnd()[axis] + 1 + offset[sides_index];
218  }
219  sides_index++;
220  } else {
221  new_strides[lengths_index] = this->getStrides()[axis];
222  new_ghost_start[lengths_index] = this->getGhostStart()[axis];
223  new_start[lengths_index] = this->getStart()[axis];
224  new_end[lengths_index] = this->getEnd()[axis];
225  new_ghost_end[lengths_index] = this->getGhostEnd()[axis];
226  lengths_index++;
227  }
228  }
229 
230  noconst_T_ptr new_data = const_cast<noconst_T_ptr>(&(*this)[first_value]); // Thunderegg doesn't care if values in ghosts are modified
231  return View<noconst_T, M>(new_data, new_strides, new_ghost_start, new_start, new_end, new_ghost_end);
232  }
