43 template <int D> class StarPatchOperator : public PatchOperator<D>
44 {
45  private:
46  constexpr int addValue(int axis) const
47  {
48  return (axis == 0) ? 0 : 1;
49  }
50  bool neumann;

60  StarPatchOperator(const Domain<D> &domain, const GhostFiller<D> &ghost_filler, bool neumann = false)
61  : PatchOperator<D>(domain, ghost_filler),
62  neumann(neumann)

77  void
78  applySinglePatch(const PatchInfo<D> &pinfo, const PatchView<const double, D> &u_view, const PatchView<double, D> &f_view, bool internal) const
79  {
80  if (internal) {
81  enforceInternalBoundaryConditions(pinfo, u_view);
82  }
83 
84  enforceBoundaryConditions(pinfo, u_view);
85 
86  std::array<double, D> h2 = pinfo.spacings;
87  for (size_t i = 0; i < D; i++) {
88  h2[i] *= h2[i];
89  }
90 
91  loop<0, D - 1>([&](int axis) {
92  int stride = u_view.getStrides()[axis];
93  loop_over_interior_indexes<D + 1>(u_view, [&](std::array<int, D + 1> coord) {
94  const double *ptr = &u_view[coord];
95  double lower = *(ptr - stride);
96  double mid = *ptr;
97  double upper = *(ptr + stride);
98  f_view[coord] = addValue(axis) * f_view[coord] + (upper - 2 * mid + lower) / h2[axis];
99  });
100  });
101  }
102  void applySinglePatch(const PatchInfo<D> &pinfo, const PatchView<const double, D> &u_view, const PatchView<double, D> &f_view) const override
103  {
104  applySinglePatch(pinfo, u_view, f_view, false);
105  }

108  const PatchView<double, D> & f_view) const override
109  {
110  applySinglePatch(pinfo, u_view, f_view, true);

112  void enforceBoundaryConditions(const PatchInfo<D> &pinfo, const PatchView<const double, D> &u_view) const
113  {
114  for (int axis = 0; axis < D; axis++) {
115  Side<D> lower_side = LowerSideOnAxis<D>(axis);
116  Side<D> upper_side = HigherSideOnAxis<D>(axis);
117  if (!pinfo.hasNbr(lower_side)) {
118  View<double, D> lower = u_view.getGhostSliceOn(lower_side, {0});
119  View<const double, D> lower_mid = u_view.getSliceOn(lower_side, {0});
120  if (neumann) {
121  loop_over_interior_indexes<D>(lower_mid, [&](std::array<int, D> coord) { lower[coord] = lower_mid[coord]; });
122  } else {
123  loop_over_interior_indexes<D>(lower_mid, [&](std::array<int, D> coord) { lower[coord] = -lower_mid[coord]; });
124  }
125  }
126  if (!pinfo.hasNbr(upper_side)) {
127  View<double, D> upper = u_view.getGhostSliceOn(upper_side, {0});
128  View<const double, D> upper_mid = u_view.getSliceOn(upper_side, {0});
129  if (neumann) {
130  loop_over_interior_indexes<D>(upper_mid, [&](std::array<int, D> coord) { upper[coord] = upper_mid[coord]; });
131  } else {
132  loop_over_interior_indexes<D>(upper_mid, [&](std::array<int, D> coord) { upper[coord] = -upper_mid[coord]; });
133  }
134  }
135  }
136  }
137  void enforceInternalBoundaryConditions(const PatchInfo<D> &pinfo, const PatchView<const double, D> &u_view) const
138  {
139  for (int axis = 0; axis < D; axis++) {
140  Side<D> lower_side = LowerSideOnAxis<D>(axis);
141  Side<D> upper_side = HigherSideOnAxis<D>(axis);
142  if (pinfo.hasNbr(lower_side)) {
143  View<double, D> lower = u_view.getGhostSliceOn(lower_side, {0});
144  View<const double, D> lower_mid = u_view.getSliceOn(lower_side, {0});
145  loop_over_interior_indexes<D>(lower_mid, [&](std::array<int, D> coord) { lower[coord] = -lower_mid[coord]; });
146  }
147  if (pinfo.hasNbr(upper_side)) {
148  View<double, D> upper = u_view.getGhostSliceOn(upper_side, {0});
149  View<const double, D> upper_mid = u_view.getSliceOn(upper_side, {0});
150  loop_over_interior_indexes<D>(upper_mid, [&](std::array<int, D> coord) { upper[coord] = -upper_mid[coord]; });
151  }
152  }
153  }
154  void modifyRHSForInternalBoundaryConditions(const PatchInfo<D> & pinfo,
155  const PatchView<const double, D> &u_view,
156  const PatchView<double, D> & f_view) const override
157  {
158  for (Side<D> s : Side<D>::getValues()) {
159  if (pinfo.hasNbr(s)) {
160  double h2 = pow(pinfo.spacings[s.getAxisIndex()], 2);
161  View<double, D> f_inner = f_view.getSliceOn(s, {0});
162  View<const double, D> u_ghost = u_view.getSliceOn(s, {-1});
163  View<const double, D> u_inner = u_view.getSliceOn(s, {0});
164  loop_over_interior_indexes<D>(f_inner,
165  [&](const std::array<int, D> &coord) { f_inner[coord] -= (u_ghost[coord] + u_inner[coord]) / h2; });
166  }
167  }
168  }
175  void addDrichletBCToRHS(Vector<D> &f, std::function<double(const std::array<double, D> &)> gfunc)
176  {
177  for (int i = 0; i < f.getNumLocalPatches(); i++) {
178  ComponentView<double, D> f_ld = f.getComponentView(0, i);
179  auto pinfo = this->getDomain().getPatchInfoVector()[i];
180  for (Side<D> s : Side<D>::getValues()) {
181  if (!pinfo.hasNbr(s)) {
182  double h2 = pow(pinfo.spacings[s.getAxisIndex()], 2);
183  View<double, D - 1> ld = f_ld.getSliceOn(s, {0});
184  nested_loop<D - 1>(ld.getStart(), ld.getEnd(), [&](const std::array<int, D - 1> &coord) {
185  std::array<double, D> real_coord;
186  DomainTools::GetRealCoordBound<D>(pinfo, coord, s, real_coord);
187  ld[coord] += -2.0 * gfunc(real_coord) / h2;
188  });
189  }
190  }
191  }
192  }
200  void addNeumannBCToRHS(Vector<D> & f,
201  std::function<double(const std::array<double, D> &)> gfunc,
202  std::array<std::function<double(const std::array<double, D> &)>, D> gfunc_grad)
203  {
204  for (int i = 0; i < f.getNumLocalPatches(); i++) {
205  ComponentView<double, D> f_ld = f.getComponentView(0, i);
206  auto pinfo = this->getDomain().getPatchInfoVector()[i];
207  for (Side<D> s : Side<D>::getValues()) {
208  if (!pinfo.hasNbr(s)) {
209  double h = pinfo.spacings[s.getAxisIndex()];
210  View<double, D - 1> ld = f_ld.getSliceOn(s, {0});
211  if (s.isLowerOnAxis()) {
212  nested_loop<D - 1>(ld.getStart(), ld.getEnd(), [&](const std::array<int, D - 1> &coord) {
213  std::array<double, D> real_coord;
214  DomainTools::GetRealCoordBound<D>(pinfo, coord, s, real_coord);
215  ld[coord] += gfunc_grad[s.getAxisIndex()](real_coord) / h;
216  });
217  } else {
218  nested_loop<D - 1>(ld.getStart(), ld.getEnd(), [&](const std::array<int, D - 1> &coord) {
219  std::array<double, D> real_coord;
220  DomainTools::GetRealCoordBound<D>(pinfo, coord, s, real_coord);
221  ld[coord] -= gfunc_grad[s.getAxisIndex()](real_coord) / h;
222  });
223  }
224  }
225  }
226  }
227  }
