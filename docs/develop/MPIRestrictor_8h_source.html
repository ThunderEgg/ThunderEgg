53  {
54  if constexpr (ENABLE_DEBUG) {

56  throw RuntimeError("fine vector is incorrect length. Expected Lenght of "

58  + std::to_string(fine.getNumLocalPatches()));
59  }
60  }
61  Vector<D> coarse = getNewCoarserVector(fine.getNumComponents());
62  Vector<D> coarse_ghost = ilc.getNewGhostVector(fine.getNumComponents());
63 
64  // fill in ghost values
65  restrictPatches(ilc.getPatchesWithGhostParent(), fine, coarse_ghost);
66 
67  // clear values in coarse vector
68  coarse.setWithGhost(0);
69 
70  // start scatter for ghost values
71  ilc.sendGhostPatchesStart(coarse, coarse_ghost);
72 
73  // fill in local values
74  restrictPatches(ilc.getPatchesWithLocalParent(), fine, coarse);
75 
76  // finish scatter for ghost values
77  ilc.sendGhostPatchesFinish(coarse, coarse_ghost);
78 
79  return coarse;
80  }
