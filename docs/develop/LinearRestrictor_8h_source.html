48  void extrapolateBoundaries(const PatchInfo<D> &pinfo, const PatchView<const double, D> &fine_view, const PatchView<double, D> &coarse_view) const
49  {
50  Orthant<D> orth = pinfo.orth_on_parent;
51  std::array<int, D> starts;
52  for (size_t i = 0; i < D; i++) {
53  starts[i] = orth.isLowerOnAxis(i) ? 0 : coarse_view.getEnd()[i] + 1;
54  }
55  // extrapolate ghost values
56  for (Side<D> s : pinfo.orth_on_parent.getExteriorSides()) {
57  // if (!pinfo.hasNbr(s)) {
58  View<const double, D> fine_ghost = fine_view.getSliceOn(s, {-1});
59  View<const double, D> fine_interior = fine_view.getSliceOn(s, {0});
60  View<double, D> coarse_ghost = coarse_view.getSliceOn(s, {-1});
61  loop_over_interior_indexes<D>(fine_ghost, [&](const std::array<int, D> &coord) {
62  std::array<int, D> coarse_coord;
63  for (size_t x = 0; x < s.getAxisIndex(); x++) {
64  coarse_coord[x] = (coord[x] + starts[x]) / 2;
65  }
66  for (size_t x = s.getAxisIndex() + 1; x < D; x++) {
67  coarse_coord[x - 1] = (coord[x - 1] + starts[x]) / 2;
68  }
69  coarse_coord[D - 1] = coord[D - 1];
70  coarse_ghost[coarse_coord] += (3 * fine_ghost[coord] - fine_interior[coord]) / (1 << D);
71  });
72  //}
73  }
74  if constexpr (D >= 2) {
75  Corner<D> c(pinfo.orth_on_parent.getIndex());
76  if (!pinfo.hasNbr(c)) {
77  std::array<int, D> neg_one;
78  neg_one.fill(-1);
79  std::array<int, D> zero;
80  zero.fill(0);
81  View<const double, 1> fine_ghost = fine_view.getSliceOn(c, neg_one);
82  View<const double, 1> fine_interior = fine_view.getSliceOn(c, zero);
83  View<double, 1> coarse_ghost = coarse_view.getSliceOn(c, neg_one);
84  loop_over_interior_indexes<1>(
85  fine_ghost, [&](const std::array<int, 1> &coord) { coarse_ghost[coord] += (3 * fine_ghost[coord] - fine_interior[coord]); });
86  }
87  }
88  }
97  void restrictToCoarserParent(const PatchInfo<D> &pinfo, int parent_index, const Vector<D> &finer_vector, Vector<D> &coarser_vector) const
98  {
99  PatchView<double, D> coarse_view = coarser_vector.getPatchView(parent_index);
100  PatchView<const double, D> fine_view = finer_vector.getPatchView(pinfo.local_index);
101  // get starting index in coarser patch
102  Orthant<D> orth = pinfo.orth_on_parent;
103  std::array<int, D> starts;
104  for (size_t i = 0; i < D; i++) {
105  starts[i] = orth.isLowerOnAxis(i) ? 0 : (coarse_view.getEnd()[i] + 1);
106  }
107 
108  // interpolate interior values
109  loop_over_interior_indexes<D + 1>(fine_view, [&](const std::array<int, D + 1> &coord) {
110  std::array<int, D + 1> coarse_coord;
111  for (size_t x = 0; x < D; x++) {
112  coarse_coord[x] = (coord[x] + starts[x]) / 2;
113  }
114  coarse_coord[D] = coord[D];
115  coarse_view[coarse_coord] += fine_view[coord] / (1 << D);
116  });
117 
118  if (extrapolate_boundary_ghosts) {
119  extrapolateBoundaries(pinfo, fine_view, coarse_view);
120  }
121  }
131  void copyToParent(const PatchInfo<D> &pinfo, int parent_index, const Vector<D> &finer_vector, Vector<D> &coarser_vector) const
132  {
133  PatchView<double, D> coarse_view = coarser_vector.getPatchView(parent_index);
134  PatchView<const double, D> fine_view = finer_vector.getPatchView(pinfo.local_index);
135  // just copy the values
136  if (extrapolate_boundary_ghosts) {
137  loop_over_all_indexes<D + 1>(fine_view, [&](const std::array<int, D + 1> &coord) { coarse_view[coord] += fine_view[coord]; });
138  } else {
139  loop_over_interior_indexes<D + 1>(fine_view, [&](const std::array<int, D + 1> &coord) { coarse_view[coord] += fine_view[coord]; });
140  }
141  }
