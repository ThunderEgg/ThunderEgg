73  void checkCoordIsInBounds(const std::array<int, D> &coord) const
74  {

172  inline void set(const std::array<int, D> &coord, T value) const
173  {
174  if constexpr (std::is_const<T>::value) {
175  if constexpr (ENABLE_DEBUG) {
176  // check that only ghost cells are being modified
177  bool is_interior_coord = true;
178  for (int i = 0; i < D; i++) {
179  is_interior_coord = is_interior_coord && (coord[i] >= start[i] && coord[i] <= end[i]);
180  }
181  if (is_interior_coord) {
182  // intertior coord
183  throw RuntimeError("interior value of const view is being modified");
184  }
185  checkCoordIsInBounds(coord);
186  }
187  int idx = 0;
188  loop<0, D - 1>([&](int i) { idx += strides[i] * coord[i]; });
189  const_cast<double *>(data)[idx] = value;
190  } else {
191  if constexpr (ENABLE_DEBUG) {
192  this->checkCoordIsInBounds(coord);
193  }
194  data[this->getIndex(coord)] = value;
195  }
196  }
