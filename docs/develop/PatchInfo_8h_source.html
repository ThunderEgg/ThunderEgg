347  template <int M> void serializeNeighbors(BufferWriter &writer) const
348  {
349  std::bitset<Face<D, M>::number_of> has_nbr;
350  for (Face<D, M> f : Face<D, M>::getValues()) {
351  has_nbr[f.getIndex()] = hasNbr(f);
352  }
353  writer << has_nbr;
354  for (Face<D, M> f : Face<D, M>::getValues()) {
355  if (hasNbr(f)) {
356  NbrType type = getNbrType(f);
357  writer << type;
358  switch (type) {
359  case NbrType::Normal: {
360  NormalNbrInfo<M> tmp = getNormalNbrInfo(f);
361  writer << tmp;
362  } break;
363  case NbrType::Fine: {
364  FineNbrInfo<M> tmp = getFineNbrInfo(f);
365  writer << tmp;
366  } break;
367  case NbrType::Coarse: {
368  CoarseNbrInfo<M> tmp = getCoarseNbrInfo(f);
369  writer << tmp;
370  } break;
371  default:
372  throw RuntimeError("Unsupported NbrType");
373  }
374  }
375  }
376  if constexpr (M > 0) {
377  serializeNeighbors<M - 1>(writer);
378  }
379  }
380  template <int M> void deserializeNeighbors(BufferReader &reader)
381  {
382  std::bitset<Face<D, M>::number_of> has_nbr;
383  reader >> has_nbr;
384  for (Face<D, M> f : Face<D, M>::getValues()) {
385  if (has_nbr[f.getIndex()]) {
386  NbrType type;
387  reader >> type;
388  NbrInfo<M> *info;
389  switch (type) {
390  case NbrType::Normal:
391  info = new NormalNbrInfo<M>();
392  reader >> *static_cast<NormalNbrInfo<M> *>(info);
393  break;
394  case NbrType::Fine:
395  info = new FineNbrInfo<M>();
396  reader >> *static_cast<FineNbrInfo<M> *>(info);
397  break;
398  case NbrType::Coarse:
399  info = new CoarseNbrInfo<M>();
400  reader >> *static_cast<CoarseNbrInfo<M> *>(info);
401  break;
402  default:
403  throw RuntimeError("Unsupported NbrType");
404  }
405 
406  setNbrInfo(f, info);
407  }
408  }
409  if constexpr (M > 0) {
410  deserializeNeighbors<M - 1>(reader);
411  }
412  }

452 template <int D> void to_json(nlohmann::json &j, const PatchInfo<D> &pinfo)
453 {
454  j["id"] = pinfo.id;
455  j["parent_id"] = pinfo.parent_id;
456  j["parent_rank"] = pinfo.parent_rank;
457  j["orth_on_parent"] = pinfo.orth_on_parent;
458  j["rank"] = pinfo.rank;
459  j["starts"] = pinfo.starts;
460  j["lengths"] = pinfo.spacings;
461  j["refine_level"] = pinfo.refine_level;
462  for (int i = 0; i < D; i++) {
463  j["lengths"][i] = pinfo.spacings[i] * pinfo.ns[i];
464  }
465  j["nbrs"] = nlohmann::json::array();
466  for (Side<D> s : Side<D>::getValues()) {
467  if (pinfo.hasNbr(s)) {
468  switch (pinfo.getNbrType(s)) {
469  case NbrType::Normal:
470  j["nbrs"].push_back(pinfo.getNormalNbrInfo(s));
471  break;
472  case NbrType::Coarse:
473  j["nbrs"].push_back(pinfo.getCoarseNbrInfo(s));
474  break;
475  case NbrType::Fine:
476  j["nbrs"].push_back(pinfo.getFineNbrInfo(s));
477  break;
478  default:
479  throw RuntimeError("Unsupported NbrType");
480  }
481  j["nbrs"].back()["type"] = pinfo.getNbrType(s);
482  j["nbrs"].back()["side"] = s;
483  }
484  }
485  if constexpr (D == 3) {
486  j["edge_nbrs"] = nlohmann::json::array();
487  for (Edge e : Edge::getValues()) {
488  if (pinfo.hasNbr(e)) {
489  switch (pinfo.getNbrType(e)) {
490  case NbrType::Normal:
491  j["edge_nbrs"].push_back(pinfo.getNormalNbrInfo(e));
492  break;
493  case NbrType::Coarse:
494  j["edge_nbrs"].push_back(pinfo.getCoarseNbrInfo(e));
495  break;
496  case NbrType::Fine:
497  j["edge_nbrs"].push_back(pinfo.getFineNbrInfo(e));
498  break;
499  default:
500  throw RuntimeError("Unsupported NbrType");
501  }
502  j["edge_nbrs"].back()["type"] = pinfo.getNbrType(e);
503  j["edge_nbrs"].back()["edge"] = e;
504  }
505  }
506  }
507  if constexpr (D >= 2) {
508  j["corner_nbrs"] = nlohmann::json::array();
509  for (Corner<D> c : Corner<D>::getValues()) {
510  if (pinfo.hasNbr(c)) {
511  switch (pinfo.getNbrType(c)) {
512  case NbrType::Normal:
513  j["corner_nbrs"].push_back(pinfo.getNormalNbrInfo(c));
514  break;
515  case NbrType::Coarse:
516  j["corner_nbrs"].push_back(pinfo.getCoarseNbrInfo(c));
517  break;
518  case NbrType::Fine:
519  j["corner_nbrs"].push_back(pinfo.getFineNbrInfo(c));
520  break;
521  default:
522  throw RuntimeError("Unsupported NbrType");
523  }
524  j["corner_nbrs"].back()["type"] = pinfo.getNbrType(c);
525  j["corner_nbrs"].back()["corner"] = c;
526  }
527  }
528  }
529  if (pinfo.child_ids[0] != -1) {
530  j["child_ids"] = pinfo.child_ids;
531  j["child_ranks"] = pinfo.child_ranks;
532  }
533 }
534 template <int D> void from_json(const nlohmann::json &j, PatchInfo<D> &pinfo)
535 {
536  pinfo.id = j["id"];
537  pinfo.parent_id = j["parent_id"];
538  pinfo.parent_rank = j["parent_rank"];
539  if (j.contains("orth_on_parent")) {
540  j["orth_on_parent"].get_to(pinfo.orth_on_parent);
541  }
542  if (j.contains("refine_level")) {
543  pinfo.refine_level = j["refine_level"];
544  }
545  pinfo.rank = j["rank"];
546  pinfo.starts = j["starts"].get<std::array<double, D>>();
547  pinfo.spacings = j["lengths"].get<std::array<double, D>>();
548  pinfo.ns.fill(1);
549  for (const auto &nbr_j : j["nbrs"]) {
550  Side<D> s = nbr_j["side"].get<Side<D>>();
551  switch (nbr_j["type"].get<NbrType>()) {
552  case NbrType::Normal:
553  pinfo.setNbrInfo(s, new NormalNbrInfo<D - 1>());
554  pinfo.getNormalNbrInfo(s) = nbr_j.get<NormalNbrInfo<D - 1>>();
555  break;
556  case NbrType::Coarse:
557  pinfo.setNbrInfo(s, new CoarseNbrInfo<D - 1>());
558  pinfo.getCoarseNbrInfo(s) = nbr_j.get<CoarseNbrInfo<D - 1>>();
559  break;
560  case NbrType::Fine:
561  pinfo.setNbrInfo(s, new FineNbrInfo<D - 1>());
562  pinfo.getFineNbrInfo(s) = nbr_j.get<FineNbrInfo<D - 1>>();
563  break;
564  default:
565  throw RuntimeError("Unsupported NbrType");
566  }
567  }
568  if constexpr (D == 3) {
569  if (j.contains("edge_nbrs")) {
570  for (const auto &nbr_j : j["edge_nbrs"]) {
571  Edge e = nbr_j["edge"].get<Edge>();
572  switch (nbr_j["type"].get<NbrType>()) {
573  case NbrType::Normal:
574  pinfo.setNbrInfo(e, new NormalNbrInfo<1>());
575  pinfo.getNormalNbrInfo(e) = nbr_j.get<NormalNbrInfo<1>>();
576  break;
577  case NbrType::Coarse:
578  pinfo.setNbrInfo(e, new CoarseNbrInfo<1>());
579  pinfo.getCoarseNbrInfo(e) = nbr_j.get<CoarseNbrInfo<1>>();
580  break;
581  case NbrType::Fine:
582  pinfo.setNbrInfo(e, new FineNbrInfo<1>());
583  pinfo.getFineNbrInfo(e) = nbr_j.get<FineNbrInfo<1>>();
584  break;
585  default:
586  throw RuntimeError("Unsupported NbrType");
587  }
588  }
589  }
590  }
591  if constexpr (D >= 2) {
592  if (j.contains("corner_nbrs")) {
593  for (const auto &nbr_j : j["corner_nbrs"]) {
594  Corner<D> c = nbr_j["corner"].get<Corner<D>>();
595  switch (nbr_j["type"].get<NbrType>()) {
596  case NbrType::Normal:
597  pinfo.setNbrInfo(c, new NormalNbrInfo<0>());
598  pinfo.getNormalNbrInfo(c) = nbr_j.get<NormalNbrInfo<0>>();
599  break;
600  case NbrType::Coarse:
601  pinfo.setNbrInfo(c, new CoarseNbrInfo<0>());
602  pinfo.getCoarseNbrInfo(c) = nbr_j.get<CoarseNbrInfo<0>>();
603  break;
604  case NbrType::Fine:
605  pinfo.setNbrInfo(c, new FineNbrInfo<0>());
606  pinfo.getFineNbrInfo(c) = nbr_j.get<FineNbrInfo<0>>();
607  break;
608  default:
609  throw RuntimeError("Unsupported NbrType");
610  }
611  }
612  }
613  }
614  if (j.contains("child_ids")) {
615  pinfo.child_ids = j["child_ids"].get<std::array<int, Orthant<D>::num_orthants>>();
616  pinfo.child_ranks = j["child_ranks"].get<std::array<int, Orthant<D>::num_orthants>>();
617  }
